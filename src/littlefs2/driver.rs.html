9/// Users of this library provide a "storage driver" by implementing this trait.
10///
11/// The `write` method is assumed to be synchronized to storage immediately.
12/// littlefs provides more flexibility - if required, this could also be exposed.
13/// Do note that due to caches, files still must be synched. And unfortunately,
14/// this can't be automatically done in `drop`, since it needs mut refs to both
15/// filesystem and storage.
16///
17/// The `*_SIZE` types must be `generic_array::typenume::consts` such as `U256`.
18///
19/// Why? Currently, associated constants can not be used (as constants...) to define
20/// arrays. This "will be fixed" as part of const generics.
21/// Once that's done, we can get rid of `generic-array`s, and replace the
22/// `*_SIZE` types with `usize`s.
23pub trait Storage {
24    // /// Error type for user-provided read/write/erase methods
25    // type Error = usize;
26
27    /// Minimum size of block read in bytes. Not in superblock
28    const READ_SIZE: usize;
29
30    /// Minimum size of block write in bytes. Not in superblock
31    const WRITE_SIZE: usize;
32
33    /// Size of an erasable block in bytes, as unsigned typenum.
34    /// Must be a multiple of both `READ_SIZE` and `WRITE_SIZE`.
35    /// [At least 128](https://github.com/littlefs-project/littlefs/issues/264#issuecomment-519963153). Stored in superblock.
36    const BLOCK_SIZE: usize;
37
38    /// Number of erasable blocks.
39    /// Hence storage capacity is `BLOCK_COUNT * BLOCK_SIZE`
40    const BLOCK_COUNT: usize;
41
42    /// Suggested values are 100-1000, higher is more performant but
43    /// less wear-leveled.  Default of -1 disables wear-leveling.
44    /// Value zero is invalid, must be positive or -1.
45    const BLOCK_CYCLES: isize = -1;
46
47    /// littlefs uses a read cache, a write cache, and one cache per per file.
48    /// Must be a multiple of `READ_SIZE` and `WRITE_SIZE`.
49    /// Must be a factor of `BLOCK_SIZE`.
50    #[allow(deprecated)]
51    type CACHE_SIZE: ArrayLength<u8>;
52
53    /// Size of the lookahead buffer used by littlefs, measured in multiples of 8 bytes.
54    #[allow(deprecated)]
55    type LOOKAHEAD_SIZE: ArrayLength<u64>;
56
57    ///// Maximum length of a filename plus one. Stored in superblock.
58    ///// Should default to 255+1, but associated type defaults don't exist currently.
59    ///// At most 1_022+1.
60    /////
61    ///// TODO: We can't actually change this - need to pass on as compile flag
62    ///// to the C backend.
63    //type FILENAME_MAX_PLUS_ONE: ArrayLength<u8>;
64
65    // /// Maximum length of a path plus one. Necessary to convert Rust string slices
66    // /// to C strings, which requires an allocation for the terminating
67    // /// zero-byte. If in doubt, set to `FILENAME_MAX_PLUS_ONE`.
68    // /// Must be larger than `FILENAME_MAX_PLUS_ONE`.
69    // type PATH_MAX_PLUS_ONE: ArrayLength<u8>;
70
71    ///// Maximum size of file. Stored in superblock.
72    ///// Defaults to 2_147_483_647 (or u31, to avoid sign issues in the C code).
73    ///// At most 2_147_483_647.
74    /////
75    ///// TODO: We can't actually change this - need to pass on as compile flag
76    ///// to the C backend.
77    //const FILEBYTES_MAX: usize = ll::LFS_FILE_MAX as _;
78
79    ///// Maximum size of custom attributes.
80    ///// Should default to 1_022, but associated type defaults don't exists currently.
81    ///// At most 1_022.
82    /////
83    ///// TODO: We can't actually change this - need to pass on as compile flag
84    ///// to the C backend.
85    //type ATTRBYTES_MAX: ArrayLength<u8>;
86
87    /// Read data from the storage device.
88    /// Guaranteed to be called only with bufs of length a multiple of READ_SIZE.
89    fn read(&mut self, off: usize, buf: &mut [u8]) -> Result<usize>;
90    /// Write data to the storage device.
91    /// Guaranteed to be called only with bufs of length a multiple of WRITE_SIZE.
92    fn write(&mut self, off: usize, data: &[u8]) -> Result<usize>;
93    /// Erase data from the storage device.
94    /// Guaranteed to be called only with bufs of length a multiple of BLOCK_SIZE.
95    fn erase(&mut self, off: usize, len: usize) -> Result<usize>;
96    // /// Synchronize writes to the storage device.
97    // fn sync(&mut self) -> Result<usize>;
98}
