9/// DER decoder.
10#[derive(Debug)]
11pub struct Decoder<'a> {
12    /// Byte slice being decoded.
13    ///
14    /// In the event an error was previously encountered this will be set to
15    /// `None` to prevent further decoding while in a bad state.
16    bytes: Option<&'a [u8]>,
17
18    /// Position within the decoded slice.
19    position: Length,
20}
21
22impl<'a> Decoder<'a> {
23    /// Create a new decoder for the given byte slice.
24    pub fn new(bytes: &'a [u8]) -> Self {
25        Self {
26            bytes: Some(bytes),
27            position: Length::ZERO,
28        }
29    }
30
31    /// Decode a value which impls the [`Decodable`] trait.
32    pub fn decode<T: Decodable<'a>>(&mut self) -> Result<T> {
33        if self.is_failed() {
34            return Err(self.error(ErrorKind::Failed));
35        }
36
37        T::decode(self).map_err(|e| {
38            self.bytes.take();
39            e.nested(self.position)
40        })
41    }
42
43    /// Return an error with the given [`ErrorKind`], annotating it with
44    /// context about where the error occurred.
45    pub fn error(&mut self, kind: ErrorKind) -> Error {
46        self.bytes.take();
47        kind.at(self.position)
48    }
49
50    /// Return an error for an invalid value with the given tag.
51    pub fn value_error(&mut self, tag: Tag) -> Error {
52        self.error(tag.value_error().kind())
53    }
54
55    /// Did the decoding operation fail due to an error?
56    pub fn is_failed(&self) -> bool {
57        self.bytes.is_none()
58    }
59
60    /// Finish decoding, returning the given value if there is no
61    /// remaining data, or an error otherwise
62    pub fn finish<T>(self, value: T) -> Result<T> {
63        if self.is_failed() {
64            Err(ErrorKind::Failed.at(self.position))
65        } else if !self.is_finished() {
66            Err(ErrorKind::TrailingData {
67                decoded: self.position,
68                remaining: self.remaining_len()?,
69            }
70            .at(self.position))
71        } else {
72            Ok(value)
73        }
74    }
75
76    /// Have we decoded all of the bytes in this [`Decoder`]?
77    ///
78    /// Returns `false` if we're not finished decoding or if a fatal error
79    /// has occurred.
80    pub fn is_finished(&self) -> bool {
81        self.remaining().map(|rem| rem.is_empty()).unwrap_or(false)
82    }
83
84    /// Attempt to decode an ASN.1 `ANY` value.
85    pub fn any(&mut self) -> Result<Any<'a>> {
86        self.decode()
87    }
88
89    /// Attempt to decode an `OPTIONAL` ASN.1 `ANY` value.
90    pub fn any_optional(&mut self) -> Result<Option<Any<'a>>> {
91        self.decode()
92    }
93
94    /// Attempt to decode ASN.1 `INTEGER` as `i8`
95    pub fn int8(&mut self) -> Result<i8> {
96        self.decode()
97    }
98
99    /// Attempt to decode ASN.1 `INTEGER` as `i16`
100    pub fn int16(&mut self) -> Result<i16> {
101        self.decode()
102    }
103
104    /// Attempt to decode unsigned ASN.1 `INTEGER` as `u8`
105    pub fn uint8(&mut self) -> Result<u8> {
106        self.decode()
107    }
108
109    /// Attempt to decode unsigned ASN.1 `INTEGER` as `u16`
110    pub fn uint16(&mut self) -> Result<u16> {
111        self.decode()
112    }
113
114    /// Attempt to decode an ASN.1 `INTEGER` as a [`UIntBytes`].
115    #[cfg(feature = "bigint")]
116    #[cfg_attr(docsrs, doc(cfg(feature = "bigint")))]
117    pub fn uint_bytes(&mut self) -> Result<UIntBytes<'a>> {
118        self.decode()
119    }
120
121    /// Attempt to decode an ASN.1 `BIT STRING`.
122    pub fn bit_string(&mut self) -> Result<BitString<'a>> {
123        self.decode()
124    }
125
126    /// Attempt to decode an ASN.1 `CONTEXT-SPECIFIC` field with the
127    /// provided [`TagNumber`].
128    ///
129    /// This method has the following behavior which is designed to simplify
130    /// handling of extension fields, which are denoted in an ASN.1 schema
131    /// using the `...` ellipsis extension marker:
132    ///
133    /// - Skips over [`ContextSpecific`] fields with a tag number lower than
134    ///   the current one, consuming and ignoring them.
135    /// - Returns `Ok(None)` if a [`ContextSpecific`] field with a higher tag
136    ///   number is encountered. These fields are not consumed in this case,
137    ///   allowing a field with a lower tag number to be omitted, then the
138    ///   higher numbered field consumed as a follow-up.
139    /// - Returns `Ok(None)` if anything other than a [`ContextSpecific`] field
140    ///   is encountered.
141    pub fn context_specific(&mut self, tag: TagNumber) -> Result<Option<Any<'a>>> {
142        loop {
143            match self.peek().map(Tag::try_from).transpose()? {
144                Some(Tag::ContextSpecific(actual_tag)) => {
145                    match actual_tag.cmp(&tag) {
146                        Ordering::Less => {
147                            // Decode and ignore lower-numbered fields if
148                            // they parse correctly.
149                            self.decode::<ContextSpecific<'_>>()?;
150                        }
151                        Ordering::Equal => {
152                            return self
153                                .decode::<ContextSpecific<'_>>()
154                                .map(|cs| Some(cs.value))
155                        }
156                        Ordering::Greater => return Ok(None),
157                    }
158                }
159                _ => return Ok(None),
160            }
161        }
162    }

203        self.decode()
204    }

208        self.decode()
209    }
210
211    /// Attempt to decode an ASN.1 `SEQUENCE`, creating a new nested
212    /// [`Decoder`] and calling the provided argument with it.
213    pub fn sequence<F, T>(&mut self, f: F) -> Result<T>
214    where
215        F: FnOnce(&mut Decoder<'a>) -> Result<T>,
216    {
217        Sequence::decode(self)?.decode_nested(f).map_err(|e| {
218            self.bytes.take();
219            e.nested(self.position)
220        })
221    }
222
223    /// Decode a single byte, updating the internal cursor.
224    pub(crate) fn byte(&mut self) -> Result<u8> {
225        match self.bytes(1u8)? {
226            [byte] => Ok(*byte),
227            _ => Err(self.error(ErrorKind::Truncated)),
228        }
229    }
230
231    /// Obtain a slice of bytes of the given length from the current cursor
232    /// position, or return an error if we have insufficient data.
233    pub(crate) fn bytes(&mut self, len: impl TryInto<Length>) -> Result<&'a [u8]> {
234        if self.is_failed() {
235            return Err(self.error(ErrorKind::Failed));
236        }
237
238        let len = len
239            .try_into()
240            .map_err(|_| self.error(ErrorKind::Overflow))?;
241
242        let result = self
243            .remaining()?
244            .get(..len.try_into()?)
245            .ok_or(ErrorKind::Truncated)?;
246
247        self.position = (self.position + len)?;
248        Ok(result)
249    }
250
251    /// Peek at the next byte in the decoder without modifying the cursor.
252    pub(crate) fn peek(&self) -> Option<u8> {
253        self.remaining()
254            .ok()
255            .and_then(|bytes| bytes.get(0).cloned())
256    }
257
258    /// Obtain the remaining bytes in this decoder from the current cursor
259    /// position.
260    fn remaining(&self) -> Result<&'a [u8]> {
261        let pos = usize::try_from(self.position)?;
262
263        self.bytes
264            .and_then(|b| b.get(pos..))
265            .ok_or_else(|| ErrorKind::Truncated.at(self.position))
266    }
267
268    /// Get the number of bytes still remaining in the buffer.
269    fn remaining_len(&self) -> Result<Length> {
270        self.remaining()?.len().try_into()
271    }
272}
273
274impl<'a> From<&'a [u8]> for Decoder<'a> {
275    fn from(bytes: &'a [u8]) -> Decoder<'a> {
276        Decoder::new(bytes)
277    }
278}

280#[cfg(test)]
281mod tests {
282    use super::Decoder;
283    use crate::{Decodable, ErrorKind, Length, Tag, TagNumber};
284    use core::convert::TryFrom;
285    use hex_literal::hex;
286
287    #[test]
288    fn context_specific_with_expected_field() {
289        let tag = TagNumber::new(0);
290
291        // Empty message
292        let mut decoder = Decoder::new(&[]);
293        assert_eq!(decoder.context_specific(tag).unwrap(), None);
294
295        // Message containing a non-context-specific type
296        let mut decoder = Decoder::new(&hex!("020100"));
297        assert_eq!(decoder.context_specific(tag).unwrap(), None);
298
299        //
300        let mut decoder = Decoder::new(&hex!("A003020100"));
301        let field = decoder.context_specific(tag).unwrap().unwrap();
302        assert_eq!(u8::try_from(field).unwrap(), 0);
303    }
304
305    #[test]
306    fn context_specific_skipping_unknown_field() {
307        let tag = TagNumber::new(1);
308        let mut decoder = Decoder::new(&hex!("A003020100A103020101"));
309        let field = decoder.context_specific(tag).unwrap().unwrap();
310        assert_eq!(u8::try_from(field).unwrap(), 1);
311    }
312
313    #[test]
314    fn context_specific_returns_none_on_greater_tag_number() {
315        let tag = TagNumber::new(0);
316        let mut decoder = Decoder::new(&hex!("A103020101"));
317        assert_eq!(decoder.context_specific(tag).unwrap(), None);
318    }
319
320    #[test]
321    fn truncated_message() {
322        let mut decoder = Decoder::new(&[]);
323        let err = bool::decode(&mut decoder).err().unwrap();
324        assert_eq!(ErrorKind::Truncated, err.kind());
325        assert_eq!(Some(Length::ZERO), err.position());
326    }
327
328    #[test]
329    fn invalid_field_length() {
330        let mut decoder = Decoder::new(&[0x02, 0x01]);
331        let err = i8::decode(&mut decoder).err().unwrap();
332        assert_eq!(ErrorKind::Length { tag: Tag::Integer }, err.kind());
333        assert_eq!(Some(Length::from(2u8)), err.position());
334    }
335
336    #[test]
337    fn trailing_data() {
338        let mut decoder = Decoder::new(&[0x02, 0x01, 0x2A, 0x00]);
339        let x = decoder.decode().unwrap();
340        assert_eq!(42i8, x);
341
342        let err = decoder.finish(x).err().unwrap();
343        assert_eq!(
344            ErrorKind::TrailingData {
345                decoded: 3u8.into(),
346                remaining: 1u8.into()
347            },
348            err.kind()
349        );
350        assert_eq!(Some(Length::from(3u8)), err.position());
351    }
