1//! Pure Rust embedded-friendly implementation of the Distinguished Encoding Rules (DER)
2//! for Abstract Syntax Notation One (ASN.1) as described in ITU [X.690].
3//!
4//! # About
5//! This crate provides a `no_std`-friendly implementation of a subset of ASN.1
6//! DER necessary for decoding/encoding various cryptography-related formats
7//! implemented as part of the [RustCrypto] project, e.g. the [`pkcs5`] and
8//! [`pkcs8`] crates.
9//!
10//! The core implementation avoids any heap usage (with convenience methods
11//! that allocate gated under the off-by-default `alloc` feature).
12//!
13//! # Minimum Supported Rust Version
14//! This crate requires **Rust 1.51** at a minimum.
15//!
16//! We may change the MSRV in the future, but it will be accompanied by a minor
17//! version bump.
18//!
19//! # Usage
20//! ## [`Decodable`] and [`Encodable`] traits
21//!
22//! The [`Decodable`] and [`Encodable`] traits are the core abstractions on
23//! which this crate is built and control what types can be (de)serialized
24//! as ASN.1 DER.
25//!
26//! The traits are impl'd for the following Rust core types:
27//!
28//! - `()`: ASN.1 `NULL` (see also [`Null`])
29//! - [`bool`]: ASN.1 `BOOLEAN`
30//! - [`i8`], [`i16`], [`i32`], [`i64`], [`i128`]: ASN.1 `INTEGER`
31//! - [`u8`], [`u16`], [`u32`], [`u64`], [`u128`]: ASN.1 `INTEGER`
32//! - [`str`], [`String`][`alloc::string::String`]: ASN.1 `UTF8String`
33//!   (see also [`Utf8String`]. `String` requires `alloc` feature)
34//! - [`BTreeSet`][`alloc::collections::BTreeSet`]: ASN.1 `SET OF` (requires `alloc` feature)
35//! - [`Option`]: ASN.1 `OPTIONAL`
36//! - [`SystemTime`][`std::time::SystemTime`]: ASN.1 `GeneralizedTime` (requires `std` feature)
37//!
38//! The following ASN.1 types provided by this crate also impl these traits:
39//!
40//! - [`Any`]: ASN.1 `ANY`
41//! - [`BitString`]: ASN.1 `BIT STRING`
42//! - [`GeneralizedTime`]: ASN.1 `GeneralizedTime`
43//! - [`Ia5String`]: ASN.1 `IA5String`
44//! - [`Null`]: ASN.1 `NULL`
45//! - [`ObjectIdentifier`]: ASN.1 `OBJECT IDENTIFIER`
46//! - [`OctetString`]: ASN.1 `OCTET STRING`
47//! - [`PrintableString`]: ASN.1 `PrintableString` (ASCII subset)
48//! - [`Sequence`]: ASN.1 `SEQUENCE`
49//! - [`SetOfRef`]: ASN.1 `SET OF`
50//! - [`UIntBytes`]: ASN.1 unsigned `INTEGER` with raw access to encoded bytes
51//! - [`UtcTime`]: ASN.1 `UTCTime`
52//! - [`Utf8String`]: ASN.1 `UTF8String`
53//!
54//! ## Example
55//! The following example implements X.509's `AlgorithmIdentifier` message type
56//! as defined in [RFC 5280 Section 4.1.1.2].
57//!
58//! The ASN.1 schema for this message type is as follows:
59//!
60//! ```text
61//! AlgorithmIdentifier  ::=  SEQUENCE  {
62//!      algorithm               OBJECT IDENTIFIER,
63//!      parameters              ANY DEFINED BY algorithm OPTIONAL  }
64//! ```
65//!
66//! Structured ASN.1 messages are typically encoded as a `SEQUENCE`, which
67//! this crate maps to a Rust struct using the [`Message`] trait. This
68//! trait is bounded on the [`Decodable`] trait and provides a blanket impl
69//! of the [`Encodable`] trait, so any type which impls [`Message`] can be
70//! used for both decoding and encoding.
71//!
72//! The [`Decoder`] and [`Encoder`] types provide the decoding/encoding API
73//! respectively, and are designed to work in conjunction with concrete ASN.1
74//! types which impl the [`Decodable`] and [`Encodable`] traits, including
75//! all types which impl the [`Message`] trait.
76//!
77//! The following code example shows how to define a struct which maps to the
78//! above schema, as well as impl the [`Message`] trait for that struct:
79//!
80//! ```
81//! # #[cfg(all(feature = "alloc", feature = "oid"))]
82//! # {
83//! // Note: the following example does not require the `std` feature at all.
84//! // It does leverage the `alloc` feature, but also provides instructions for
85//! // "heapless" usage when the `alloc` feature is disabled.
86//! use core::convert::{TryFrom, TryInto};
87//! use der::{
88//!     asn1::{Any, ObjectIdentifier},
89//!     Decodable, Encodable, Message
90//! };
91//!
92//! /// X.509 `AlgorithmIdentifier`.
93//! #[derive(Copy, Clone, Debug, Eq, PartialEq)]
94//! pub struct AlgorithmIdentifier<'a> {
95//!     /// This field contains an ASN.1 `OBJECT IDENTIFIER`, a.k.a. OID.
96//!     pub algorithm: ObjectIdentifier,
97//!
98//!     /// This field is `OPTIONAL` and contains the ASN.1 `ANY` type, which
99//!     /// in this example allows arbitrary algorithm-defined parameters.
100//!     pub parameters: Option<Any<'a>>
101//! }
102//!
103//! // Note: types which impl `TryFrom<Any<'a>, Error = der::Error>` receive a
104//! // blanket impl of the `Decodable` trait, therefore satisfying the
105//! // `Decodable` trait bounds on `Message`, which is impl'd below.
106//! impl<'a> TryFrom<Any<'a>> for AlgorithmIdentifier<'a> {
107//!    type Error = der::Error;
108//!
109//!     fn try_from(any: Any<'a>) -> der::Result<AlgorithmIdentifier> {
110//!         // The `Any::sequence` method asserts that an `Any` value
111//!         // contains an ASN.1 `SEQUENCE` then calls the provided `FnOnce`
112//!         // with a `der::Decoder` which can be used to decode it.
113//!         any.sequence(|decoder| {
114//!             // The `der::Decoder::Decode` method can be used to decode any
115//!             // type which impls the `Decodable` trait, which is impl'd for
116//!             // all of the ASN.1 built-in types in the `der` crate.
117//!             //
118//!             // Note that if your struct's fields don't contain an ASN.1
119//!             // built-in type specifically, there are also helper methods
120//!             // for all of the built-in types supported by this library
121//!             // which can be used to select a specific type.
122//!             //
123//!             // For example, another way of decoding this particular field,
124//!             // which contains an ASN.1 `OBJECT IDENTIFIER`, is by calling
125//!             // `decoder.oid()`. Similar methods are defined for other
126//!             // ASN.1 built-in types.
127//!             let algorithm = decoder.decode()?;
128//!
129//!             // This field contains an ASN.1 `OPTIONAL` type. The `der` crate
130//!             // maps this directly to Rust's `Option` type and provides
131//!             // impls of the `Decodable` and `Encodable` traits for `Option`.
132//!             // To explicitly request an `OPTIONAL` type be decoded, use the
133//!             // `decoder.optional()` method.
134//!             let parameters = decoder.decode()?;
135//!
136//!             // The value returned from the provided `FnOnce` will be
137//!             // returned from the `any.sequence(...)` call above.
138//!             // Note that the entire sequence body *MUST* be consumed
139//!             // or an error will be returned.
140//!             Ok(Self { algorithm, parameters })
141//!         })
142//!     }
143//! }
144//!
145//! impl<'a> Message<'a> for AlgorithmIdentifier<'a> {
146//!     // The `Message::fields` method is used for encoding and functions as
147//!     // a visitor for all of the fields in a message.
148//!     //
149//!     // To implement it, you must define a slice containing `Encodable`
150//!     // trait objects, then pass it to the provided `field_encoder`
151//!     // function, which is implemented by the `der` crate and handles
152//!     // message serialization.
153//!     //
154//!     // Trait objects are used because they allow for slices containing
155//!     // heterogeneous field types, and a callback is used to allow for the
156//!     // construction of temporary field encoder types. The latter means
157//!     // that the fields of your Rust struct don't necessarily need to
158//!     // impl the `Encodable` trait, but if they don't you must construct
159//!     // a temporary wrapper value which does.
160//!     //
161//!     // Types which impl the `Message` trait receive blanket impls of both
162//!     // the `Encodable` and `Tagged` traits (where the latter is impl'd as
163//!     // `Tagged::TAG = der::Tag::Sequence`.
164//!     fn fields<F, T>(&self, field_encoder: F) -> der::Result<T>
165//!     where
166//!         F: FnOnce(&[&dyn Encodable]) -> der::Result<T>,
167//!     {
168//!         field_encoder(&[&self.algorithm, &self.parameters])
169//!     }
170//! }
171//!
172//! // Example parameters value: OID for the NIST P-256 elliptic curve.
173//! let parameters = "1.2.840.10045.3.1.7".parse::<ObjectIdentifier>().unwrap();
174//!
175//! // We need to convert `parameters` into an `Any<'a>` type, which wraps a
176//! // `&'a [u8]` byte slice.
177//! //
178//! // To do that, we need owned DER-encoded data so that we can have
179//! // `Any` borrow a reference to it, so we have to serialize the OID.
180//! //
181//! // When the `alloc` feature of this crate is enabled, any type that impls
182//! // the `Encodable` trait including all ASN.1 built-in types and any type
183//! // which impls `Message` can be serialized by calling `Encodable::to_vec()`.
184//! //
185//! // If you would prefer to avoid allocations, you can create a byte array
186//! // as backing storage instead, pass that to `der::Encoder::new`, and then
187//! // encode the `parameters` value using `encoder.encode(parameters)`.
188//! let der_encoded_parameters = parameters.to_vec().unwrap();
189//!
190//! let algorithm_identifier = AlgorithmIdentifier {
191//!     // OID for `id-ecPublicKey`, if you're curious
192//!     algorithm: "1.2.840.10045.2.1".parse().unwrap(),
193//!
194//!     // `Any<'a>` impls `TryFrom<&'a [u8]>`, which parses the provided
195//!     // slice as an ASN.1 DER-encoded message.
196//!     parameters: Some(der_encoded_parameters.as_slice().try_into().unwrap())
197//! };
198//!
199//! // Serialize the `AlgorithmIdentifier` created above as ASN.1 DER,
200//! // allocating a `Vec<u8>` for storage.
201//! //
202//! // As mentioned earlier, if you don't have the `alloc` feature enabled you
203//! // can create a fix-sized array instead, then call `Encoder::new` with a
204//! // reference to it, then encode the message using
205//! // `encoder.encode(algorithm_identifier)`, then finally `encoder.finish()`
206//! // to obtain a byte slice containing the encoded message.
207//! let der_encoded_algorithm_identifier = algorithm_identifier.to_vec().unwrap();
208//!
209//! // Deserialize the `AlgorithmIdentifier` we just serialized from ASN.1 DER
210//! // using `der::Decodable::from_bytes`.
211//! let decoded_algorithm_identifier = AlgorithmIdentifier::from_der(
212//!     &der_encoded_algorithm_identifier
213//! ).unwrap();
214//!
215//! // Ensure the original `AlgorithmIdentifier` is the same as the one we just
216//! // decoded from ASN.1 DER.
217//! assert_eq!(algorithm_identifier, decoded_algorithm_identifier);
218//! # }
219//! ```
220//!
221//! ## Custom derive support
222//! When the `derive` feature of this crate is enabled, the following custom
223//! derive macros are available:
224//!
225//! - [`Choice`]: derive for `CHOICE` enum (see [`der_derive::Choice`])
226//! - [`Message`]: derive for `SEQUENCE` struct (see [`der_derive::Message`])
227//!
228//! ### Derive [`Message`] for `SEQUENCE` struct
229//! The following is a code example of how to use the [`Message`] custom derive:
230//!
231//! ```
232//! # #[cfg(all(feature = "alloc", feature = "derive", feature = "oid"))]
233//! # {
234//! use der::{asn1::{Any, ObjectIdentifier}, Encodable, Decodable, Message};
235//! use core::convert::TryInto;
236//!
237//! /// X.509 `AlgorithmIdentifier` (same as above)
238//! #[derive(Copy, Clone, Debug, Eq, PartialEq, Message)] // NOTE: added `Message`
239//! pub struct AlgorithmIdentifier<'a> {
240//!     /// This field contains an ASN.1 `OBJECT IDENTIFIER`, a.k.a. OID.
241//!     pub algorithm: ObjectIdentifier,
242//!
243//!     /// This field is `OPTIONAL` and contains the ASN.1 `ANY` type, which
244//!     /// in this example allows arbitrary algorithm-defined parameters.
245//!     pub parameters: Option<Any<'a>>
246//! }
247//!
248//! // Example parameters value: OID for the NIST P-256 elliptic curve.
249//! let parameters = "1.2.840.10045.3.1.7".parse::<ObjectIdentifier>().unwrap();
250//! let der_encoded_parameters = parameters.to_vec().unwrap();
251//!
252//! let algorithm_identifier = AlgorithmIdentifier {
253//!     // OID for `id-ecPublicKey`, if you're curious
254//!     algorithm: "1.2.840.10045.2.1".parse().unwrap(),
255//!
256//!     // `Any<'a>` impls `TryFrom<&'a [u8]>`, which parses the provided
257//!     // slice as an ASN.1 DER-encoded message.
258//!     parameters: Some(der_encoded_parameters.as_slice().try_into().unwrap())
259//! };
260//!
261//! // Encode
262//! let der_encoded_algorithm_identifier = algorithm_identifier.to_vec().unwrap();
263//!
264//! // Decode
265//! let decoded_algorithm_identifier = AlgorithmIdentifier::from_der(
266//!     &der_encoded_algorithm_identifier
267//! ).unwrap();
268//!
269//! assert_eq!(algorithm_identifier, decoded_algorithm_identifier);
270//! # }
271//! ```
272//!
273//! For fields which don't directly impl [`Decodable`] and [`Encodable`],
274//! you can add annotations to convert to an intermediate ASN.1 type
275//! first, so long as that type impls `TryFrom` and `Into` for the
276//! ASN.1 type.
277//!
278//! For example, structs containing `&'a [u8]` fields may want them encoded
279//! as either a `BIT STRING` or `OCTET STRING`. By using the
280//! `#[asn1(type = "BIT STRING")]` annotation it's possible to select which
281//! ASN.1 type should be used.
282//!
283//! Building off the above example:
284//!
285//! ```rust
286//! # #[cfg(all(feature = "alloc", feature = "derive", feature = "oid"))]
287//! # {
288//! # use der::{asn1::{Any, ObjectIdentifier}, Message};
289//! #
290//! # #[derive(Copy, Clone, Debug, Eq, PartialEq, Message)]
291//! # pub struct AlgorithmIdentifier<'a> {
292//! #     pub algorithm: ObjectIdentifier,
293//! #     pub parameters: Option<Any<'a>>
294//! # }
295//! /// X.509 `SubjectPublicKeyInfo` (SPKI)
296//! #[derive(Copy, Clone, Debug, Eq, PartialEq, Message)]
297//! pub struct SubjectPublicKeyInfo<'a> {
298//!     /// X.509 `AlgorithmIdentifier`
299//!     pub algorithm: AlgorithmIdentifier<'a>,
300//!
301//!     /// Public key data
302//!     #[asn1(type = "BIT STRING")]
303//!     pub subject_public_key: &'a [u8],
304//! }
305//! # }
306//! ```
