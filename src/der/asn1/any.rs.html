12/// ASN.1 `ANY`: represents any explicitly tagged ASN.1 value.
13///
14/// Technically `ANY` hasn't been a recommended part of ASN.1 since the X.209
15/// revision from 1988. It was deprecated and replaced by Information Object
16/// Classes in X.680 in 1994, and X.690 no longer refers to it whatsoever.
17///
18/// Nevertheless, this crate defines an [`Any`] type as it remains a familiar
19/// and useful concept which is still extensively used in things like
20/// PKI-related RFCs.
21#[derive(Copy, Clone, Debug, Eq, PartialEq, PartialOrd, Ord)]
22pub struct Any<'a> {
23    /// Tag representing the type of the encoded value.
24    tag: Tag,
25
26    /// Encoded length of this [`Any`] value.
27    length: Length,
28
29    /// Inner value encoded as bytes.
30    value: ByteSlice<'a>,
31}
32
33impl<'a> Any<'a> {
34    /// Create a new [`Any`] from the provided [`Tag`] and byte slice.
35    pub fn new(tag: Tag, bytes: &'a [u8]) -> Result<Self> {
36        let value = ByteSlice::new(bytes).map_err(|_| ErrorKind::Length { tag })?;
37
38        let length = if has_leading_zero_byte(tag) {
39            (value.len() + 1u8)?
40        } else {
41            value.len()
42        };
43
44        Ok(Self { tag, length, value })
45    }
46
47    /// Infallible creation of an [`Any`] from a [`ByteSlice`].
48    pub(crate) fn from_tag_and_value(tag: Tag, value: ByteSlice<'a>) -> Self {
49        Self {
50            tag,
51            length: value.len(),
52            value,
53        }
54    }
55
56    /// Get the tag for this [`Any`] type.
57    pub fn tag(self) -> Tag {
58        self.tag
59    }
60
61    /// Get the [`Length`] of this [`Any`] type's value.
62    pub fn len(self) -> Length {
63        self.length
64    }
65
66    /// Is the body of this [`Any`] type empty?
67    pub fn is_empty(self) -> bool {
68        self.value.is_empty()
69    }
70
71    /// Is this value an ASN.1 NULL value?
72    pub fn is_null(self) -> bool {
73        Null::try_from(self).is_ok()
74    }
75
76    /// Get the raw value for this [`Any`] type as a byte slice.
77    pub fn as_bytes(self) -> &'a [u8] {
78        self.value.as_bytes()
79    }
80
81    /// Attempt to decode an ASN.1 `BIT STRING`.
82    pub fn bit_string(self) -> Result<BitString<'a>> {
83        self.try_into()
84    }
85
86    /// Attempt to decode an ASN.1 `CONTEXT-SPECIFIC` field.
87    pub fn context_specific(self) -> Result<ContextSpecific<'a>> {
88        self.try_into()
89    }
90
91    /// Attempt to decode an ASN.1 `GeneralizedTime`.
92    pub fn generalized_time(self) -> Result<GeneralizedTime> {
93        self.try_into()
94    }
95
96    /// Attempt to decode an ASN.1 `IA5String`.
97    pub fn ia5_string(self) -> Result<Ia5String<'a>> {
98        self.try_into()
99    }
100
101    /// Attempt to decode an ASN.1 `OCTET STRING`.
102    pub fn octet_string(self) -> Result<OctetString<'a>> {
103        self.try_into()
104    }
105
106    /// Attempt to decode an ASN.1 `OBJECT IDENTIFIER`.
107    #[cfg(feature = "oid")]
108    #[cfg_attr(docsrs, doc(cfg(feature = "oid")))]
109    pub fn oid(self) -> Result<ObjectIdentifier> {
110        self.try_into()
111    }
112
113    /// Attempt to decode an ASN.1 `OPTIONAL` value.
114    pub fn optional<T>(self) -> Result<Option<T>>
115    where
116        T: Choice<'a> + TryFrom<Self, Error = Error>,
117    {
118        if T::can_decode(self.tag) {
119            T::try_from(self).map(Some)
120        } else {
121            Ok(None)
122        }
123    }
124
125    /// Attempt to decode an ASN.1 `PrintableString`.
126    pub fn printable_string(self) -> Result<PrintableString<'a>> {
127        self.try_into()
128    }
129
130    /// Attempt to decode this value an ASN.1 `SEQUENCE`, creating a new
131    /// nested [`Decoder`] and calling the provided argument with it.
132    pub fn sequence<F, T>(self, f: F) -> Result<T>
133    where
134        F: FnOnce(&mut Decoder<'a>) -> Result<T>,
135    {
136        Sequence::try_from(self)?.decode_nested(f)
137    }
138
139    /// Attempt to decode an ASN.1 `UTCTime`.
140    pub fn utc_time(self) -> Result<UtcTime> {
141        self.try_into()
142    }
143
144    /// Attempt to decode an ASN.1 `UTF8String`.
145    pub fn utf8_string(self) -> Result<Utf8String<'a>> {
146        self.try_into()
147    }
148}

150impl<'a> Choice<'a> for Any<'a> {
151    fn can_decode(_: Tag) -> bool {
152        true
153    }
154}
155
156impl<'a> Decodable<'a> for Any<'a> {
157    fn decode(decoder: &mut Decoder<'a>) -> Result<Any<'a>> {
158        let header = Header::decode(decoder)?;
159        let tag = header.tag;
160        let mut value = decoder
161            .bytes(header.length)
162            .map_err(|_| decoder.error(ErrorKind::Length { tag }))?;
163
164        if has_leading_zero_byte(tag) {
165            let (byte, rest) = value
166                .split_first()
167                .ok_or(ErrorKind::Truncated)
168                .map_err(|e| decoder.error(e))?;
169
170            // The first octet of a BIT STRING encodes the number of unused bits.
171            // We presently constrain this to 0.
172            if *byte != 0 {
173                return Err(decoder.error(ErrorKind::Noncanonical { tag }));
174            }
175
176            value = rest;
177        }
178
179        Self::new(tag, value).map_err(|e| decoder.error(e.kind()))
180    }
181}
182
183impl<'a> Encodable for Any<'a> {
184    fn encoded_len(&self) -> Result<Length> {
185        self.len().for_tlv()
186    }
187
188    fn encode(&self, encoder: &mut Encoder<'_>) -> Result<()> {
189        Header::new(self.tag, self.len())?.encode(encoder)?;
190
191        if has_leading_zero_byte(self.tag) {
192            encoder.byte(0)?;
193        }
194
195        encoder.bytes(self.as_bytes())
196    }
197}

199impl<'a> TryFrom<&'a [u8]> for Any<'a> {
200    type Error = Error;
201
202    fn try_from(bytes: &'a [u8]) -> Result<Any<'a>> {
203        Any::from_der(bytes)
204    }

207// Special handling for the leading `0` byte on [`BitString`]
208impl<'a> TryFrom<Any<'a>> for BitString<'a> {
209    type Error = Error;
210
211    fn try_from(any: Any<'a>) -> Result<BitString<'a>> {
212        any.tag().assert_eq(Tag::BitString)?;
213
214        Ok(BitString {
215            inner: any.value,
216            encoded_len: any.length,
217        })
218    }
219}

221// Special handling for the leading `0` byte on [`BitString`]
222impl<'a> From<BitString<'a>> for Any<'a> {
223    fn from(bit_string: BitString<'a>) -> Any<'a> {
224        Any {
225            tag: Tag::BitString,
226            length: bit_string.encoded_len,
227            value: bit_string.inner,
228        }
229    }
230}

232/// Does a value with this tag have a leading zero byte?
233///
234/// This is mostly a hack for `BIT STRING`, and permits simple `From`
235/// conversions from `BitString` into `Any`.
236// TODO(tarcieri): better generalize this? or is there a better solution?
237fn has_leading_zero_byte(tag: Tag) -> bool {
238    tag == Tag::BitString
