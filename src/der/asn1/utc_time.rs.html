13/// Maximum duration since `UNIX_EPOCH` which can be represented as a `UTCTime`
14/// (non-inclusive) according to RFC 5280 rules.
15///
16/// This corresponds to the RFC3339 date: `2050-01-01T00:00:00Z`
17const MAX_UNIX_DURATION: Duration = Duration::from_secs(2_524_608_000);
18
19/// ASN.1 `UTCTime` type.
20///
21/// This type implements the validity requirements specified in
22/// [RFC 5280 Section 4.1.2.5.1][1], namely:
23///
24/// > For the purposes of this profile, UTCTime values MUST be expressed in
25/// > Greenwich Mean Time (Zulu) and MUST include seconds (i.e., times are
26/// > `YYMMDDHHMMSSZ`), even where the number of seconds is zero.  Conforming
27/// > systems MUST interpret the year field (`YY`) as follows:
28/// >
29/// > - Where `YY` is greater than or equal to 50, the year SHALL be
30/// >   interpreted as `19YY`; and
31/// > - Where `YY` is less than 50, the year SHALL be interpreted as `20YY`.
32///
33/// [1]: https://tools.ietf.org/html/rfc5280#section-4.1.2.5.1
34#[derive(Copy, Clone, Debug, Eq, PartialEq, PartialOrd, Ord)]
35pub struct UtcTime(Duration);
36
37impl UtcTime {
38    /// Length of an RFC 5280-flavored ASN.1 DER-encoded [`UtcTime`].
39    pub const LENGTH: Length = Length::new(13);
40
41    /// Create a new [`UtcTime`] given a [`Duration`] since `UNIX_EPOCH`
42    /// (a.k.a. "Unix time")
43    pub fn new(unix_duration: Duration) -> Result<Self> {
44        if unix_duration < MAX_UNIX_DURATION {
45            Ok(Self(unix_duration))
46        } else {
47            Err(Self::TAG.value_error())
48        }
49    }
50
51    /// Get the duration of this timestamp since `UNIX_EPOCH`.
52    pub fn unix_duration(&self) -> Duration {
53        self.0
54    }
55
56    /// Instantiate from [`SystemTime`].
57    #[cfg(feature = "std")]
58    #[cfg_attr(docsrs, doc(cfg(feature = "std")))]
59    pub fn from_system_time(time: SystemTime) -> Result<Self> {
60        time.duration_since(UNIX_EPOCH)
61            .map_err(|_| Self::TAG.value_error())
62            .and_then(Self::new)
63    }
64
65    /// Convert to [`SystemTime`].
66    #[cfg(feature = "std")]
67    #[cfg_attr(docsrs, doc(cfg(feature = "std")))]
68    pub fn to_system_time(&self) -> SystemTime {
69        UNIX_EPOCH + self.unix_duration()
70    }
71}

87impl TryFrom<Any<'_>> for UtcTime {
88    type Error = Error;
89
90    fn try_from(any: Any<'_>) -> Result<UtcTime> {
91        any.tag().assert_eq(Self::TAG)?;
92
93        match *any.as_bytes() {
94            // RFC 5280 requires mandatory seconds and Z-normalized time zone
95            [year1, year2, mon1, mon2, day1, day2, hour1, hour2, min1, min2, sec1, sec2, b'Z'] => {
96                let year = datetime::decode_decimal(Self::TAG, year1, year2)?;
97                let month = datetime::decode_decimal(Self::TAG, mon1, mon2)?;
98                let day = datetime::decode_decimal(Self::TAG, day1, day2)?;
99                let hour = datetime::decode_decimal(Self::TAG, hour1, hour2)?;
100                let minute = datetime::decode_decimal(Self::TAG, min1, min2)?;
101                let second = datetime::decode_decimal(Self::TAG, sec1, sec2)?;
102
103                // RFC 5280 rules for interpreting the year
104                let year = if year >= 50 { year + 1900 } else { year + 2000 };
105
106                DateTime::new(year, month, day, hour, minute, second)
107                    .and_then(|dt| dt.unix_duration())
108                    .ok_or_else(|| Self::TAG.value_error())
109                    .and_then(Self::new)
110            }
111            _ => Err(Self::TAG.value_error()),
112        }
113    }
114}

116impl Encodable for UtcTime {
117    fn encoded_len(&self) -> Result<Length> {
118        Self::LENGTH.for_tlv()
119    }
120
121    fn encode(&self, encoder: &mut Encoder<'_>) -> Result<()> {
122        Header::new(Self::TAG, Self::LENGTH)?.encode(encoder)?;
123
124        let datetime =
125            DateTime::from_unix_duration(self.0).ok_or_else(|| Self::TAG.value_error())?;
126
127        debug_assert!((1950..2050).contains(&datetime.year()));
128
129        let year = match datetime.year() {
130            y @ 1950..=1999 => y - 1900,
131            y @ 2000..=2049 => y - 2000,
132            _ => return Err(Self::TAG.value_error()),
133        };
134
135        datetime::encode_decimal(encoder, Self::TAG, year)?;
136        datetime::encode_decimal(encoder, Self::TAG, datetime.month())?;
137        datetime::encode_decimal(encoder, Self::TAG, datetime.day())?;
138        datetime::encode_decimal(encoder, Self::TAG, datetime.hour())?;
139        datetime::encode_decimal(encoder, Self::TAG, datetime.minute())?;
140        datetime::encode_decimal(encoder, Self::TAG, datetime.second())?;
141        encoder.byte(b'Z')
142    }
143}

149#[cfg(test)]
150mod tests {
151    use super::{DateTime, UtcTime};
152    use crate::{Decodable, Encodable, Encoder};
153    use hex_literal::hex;
154
155    #[test]
156    fn round_trip_vector() {
157        let example_bytes = hex!("17 0d 39 31 30 35 30 36 32 33 34 35 34 30 5a");
158        let utc_time = UtcTime::from_der(&example_bytes).unwrap();
159        assert_eq!(utc_time.unix_duration().as_secs(), 673573540);
160
161        let mut buf = [0u8; 128];
162        let mut encoder = Encoder::new(&mut buf);
163        utc_time.encode(&mut encoder).unwrap();
164        assert_eq!(example_bytes, encoder.finish().unwrap());
165    }
166
167    #[test]
168    fn round_trip_examples() {
169        for year in 1970..=2049 {
170            for month in 1..=12 {
171                let max_day = if month == 2 { 28 } else { 30 };
172
173                for day in 1..=max_day {
174                    for hour in 0..=23 {
175                        let datetime1 = DateTime::new(year, month, day, hour, 0, 0).unwrap();
176                        let utc_time1 = UtcTime::new(datetime1.unix_duration().unwrap()).unwrap();
177
178                        let mut buf = [0u8; 128];
179                        let mut encoder = Encoder::new(&mut buf);
180                        utc_time1.encode(&mut encoder).unwrap();
181                        let der_bytes = encoder.finish().unwrap();
182
183                        let utc_time2 = UtcTime::from_der(der_bytes).unwrap();
184                        assert_eq!(utc_time1, utc_time2);
185                    }
186                }
187            }
188        }
189    }
