1//! Support for encoding negative integers
2
3use super::is_highest_bit_set;
4use crate::{asn1::Any, Encodable, Encoder, Header, Length, Result, Tag};
5use core::convert::TryFrom;
6
7/// Decode an unsigned integer of the specified size.
8///
9/// Returns a byte array of the requested size containing a big endian integer.
10pub(super) fn decode_array<const N: usize>(any: Any<'_>) -> Result<[u8; N]> {
11    any.tag().assert_eq(Tag::Integer)?;
12    let mut output = [0xFFu8; N];
13    let offset = N.saturating_sub(any.as_bytes().len());
14    output[offset..].copy_from_slice(any.as_bytes());
15    Ok(output)
16}
17
18/// Encode the given big endian bytes representing an integer as ASN.1 DER.
19pub(super) fn encode(encoder: &mut Encoder<'_>, bytes: &[u8]) -> Result<()> {
20    let bytes = strip_leading_ones(&bytes);
21    let len = Length::try_from(bytes.len())?;
22    Header::new(Tag::Integer, len)?.encode(encoder)?;
23    encoder.bytes(bytes)
24}
25
26/// Get the encoded length for the given unsigned integer serialized as bytes.
27#[inline]
28pub(super) fn encoded_len(bytes: &[u8]) -> Result<Length> {
29    Length::try_from(strip_leading_ones(&bytes).len())
30}
31
32/// Strip the leading all-ones bytes from the given byte slice.
33fn strip_leading_ones(mut bytes: &[u8]) -> &[u8] {
34    while let Some((byte, rest)) = bytes.split_first() {
35        if *byte == 0xFF && is_highest_bit_set(rest) {
36            bytes = rest;
37            continue;
38        }
39
40        break;
41    }
42
43    bytes
44}
