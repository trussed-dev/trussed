12/// ASN.1 `SEQUENCE` type.
13#[derive(Copy, Clone, Debug, Eq, PartialEq, PartialOrd, Ord)]
14pub struct Sequence<'a> {
15    /// Inner value
16    inner: ByteSlice<'a>,
17}
18
19impl<'a> Sequence<'a> {
20    /// Create a new [`Sequence`] from a slice.
21    pub(crate) fn new(slice: &'a [u8]) -> Result<Self> {
22        ByteSlice::new(slice)
23            .map(|inner| Self { inner })
24            .map_err(|_| ErrorKind::Length { tag: Self::TAG }.into())
25    }
26
27    /// Borrow the inner byte sequence.
28    pub fn as_bytes(&self) -> &'a [u8] {
29        self.inner.as_bytes()
30    }
31
32    /// Decode values nested within a sequence, creating a new [`Decoder`] for
33    /// the data contained in the sequence's body and passing it to the provided
34    /// [`FnOnce`].
35    pub fn decode_nested<F, T>(&self, f: F) -> Result<T>
36    where
37        F: FnOnce(&mut Decoder<'a>) -> Result<T>,
38    {
39        let mut seq_decoder = Decoder::new(self.as_bytes());
40        let result = f(&mut seq_decoder)?;
41        seq_decoder.finish(result)
42    }
43
44    /// Iterate over the values in a heterogenously typed sequence.
45    pub fn iter<T: Decodable<'a>>(&self) -> SequenceIter<'a, T> {
46        SequenceIter::new(Decoder::new(self.as_bytes()))
47    }
48}
49
50impl AsRef<[u8]> for Sequence<'_> {
51    fn as_ref(&self) -> &[u8] {
52        self.as_bytes()
53    }
54}
55
56impl<'a> TryFrom<Any<'a>> for Sequence<'a> {
57    type Error = Error;
58
59    fn try_from(any: Any<'a>) -> Result<Self> {
60        any.tag().assert_eq(Tag::Sequence)?;
61        Self::new(any.as_bytes())
62    }
63}
64
65impl<'a> From<Sequence<'a>> for Any<'a> {
66    fn from(seq: Sequence<'a>) -> Any<'a> {
67        Any::from_tag_and_value(Tag::Sequence, seq.inner)
68    }
69}
70
71impl<'a> Encodable for Sequence<'a> {
72    fn encoded_len(&self) -> Result<Length> {
73        Any::from(*self).encoded_len()
74    }
75
76    fn encode(&self, encoder: &mut Encoder<'_>) -> Result<()> {
77        Any::from(*self).encode(encoder)
78    }
79}
80
81impl<'a> Tagged for Sequence<'a> {
82    const TAG: Tag = Tag::Sequence;
