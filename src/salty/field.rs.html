11/// Requirements on an implementation of the base field.
12///
13/// There are *many* ways to implement field arithmetic in
14/// the base field of integers modulo 2**255 - 19.
15///
16/// This trait specifies our requirements, such that end users
17/// can experiment with their own ideas.
18///
19/// This crate, as of now, offers two implementations:
20/// - TweetNaCl: a transliteration of the TweetNaCl code to Rust
21/// - Haase: a fast implementation in assembly, due to Bjoern Haase
22///
23/// Planned: Schoolbook: our own attempt at a fast yet readable implementation
24///
25/// Originally, the plan was to have everything generic over the field
26/// implementation, so far we have not been successful in convincing the Rust
27/// compiler of this. Therefore, currently the implementations must be selected
28/// at compile time using feature flags.
29pub trait FieldImplementation
30where
31    // Self: Sized,
32    // Self: Clone,
33    Self: Copy,
34    // Self: !Copy,  // not sure - curve25519-dalek uses implicit copies, do we want this?
35    Self: Debug,
36    // Self: Default,  // would want this to return zero I think
37
38    // for<'a, 'b> &'a Self: PartialEq
39    Self: ConditionallySelectable,
40    for<'b> Self: ConstantTimeEq,
41
42    Self: PartialEq,
43
44    for<'a, 'b> &'a Self: Add<&'b Self, Output = Self>,
45    for<'b> Self: AddAssign<&'b Self>,
46
47    for<'a> &'a Self: Neg<Output = Self>,
48
49    for<'a, 'b> &'a Self: Sub<&'b Self, Output = Self>,
50    for<'b> Self: SubAssign<&'b Self>,
51
52    for<'a, 'b> &'a Self: Mul<&'b Self, Output = Self>,
53    for<'b> Self: MulAssign<&'b Self>,
54{
55    /// Internal representation as limbs
56    type Limbs;
57
58    // TODO: maybe have statics outside,
59    // and demand functions returning &'static Self instead?
60    const ZERO: Self;
61    const ONE: Self;
62    const D: Self;
63    const D2: Self;
64    const APLUS2_OVER_FOUR: Self;
65    const EDWARDS_BASEPOINT_X: Self;
66    const EDWARDS_BASEPOINT_Y: Self;
67    const I: Self;
68    const MONTGOMERY_BASEPOINT_U: Self;
69
70    // /// swap p and q iff b is true, in constant time
71    // // TODO: would be great to mark this with an attribute
72    // // like #[constant_time], and have this picked up by a testing
73    // // harness, that actually tests this!
74    // pub fn conditional_swap(p: &mut FieldElement, q: &mut FieldElement, b: bool);
75
76    // fn reduce(&mut self);
77
78    // /// We don't want to introduce Copy on all our types,
79    // /// and be indirect about swap, so we do this ourselves
80    // fn conditional_swap(&mut self, other: &mut Self, b: bool);
81
82    /// to canonical representation as little-endian bytes
83    fn to_bytes(&self) -> [u8; 32];
84
85    /// construct from possibly non-canonical representation as little-endian bytes
86    fn from_unreduced_bytes(bytes: &[u8; 32]) -> Self {
87        let unreduced = Self::from_bytes_unchecked(bytes);
88        Self::from_bytes_unchecked(&unreduced.to_bytes())
89    }
90
91    /// construct from canonical representation as little-endian bytes
92    fn from_bytes_unchecked(bytes: &[u8; 32]) -> Self;
93
94    /// construct from canonical representation as little-endian bytes, with validity check
95    fn from_bytes(bytes: &[u8; 32]) -> Result<Self> {
96        // TODO: convert this into a TryFrom
97        let unchecked = Self::from_bytes_unchecked(bytes);
98        let canonical_representation = unchecked.to_bytes();
99        if bool::from(bytes.ct_eq(&canonical_representation)) {
100            Ok(unchecked)
101        } else {
102            Err(Error::NonCanonicalFieldElement)
103        }
104    }
105
106    /// parity of field element, viewed as integer modulo 2**255 - 19
107    fn parity(&self) -> u8 {
108        let d = self.to_bytes();
109        d[0] & 1
110    }
111
112    /// default implementation, actual implementation may override
113    /// this with a faster version
114    fn squared(&self) -> Self {
115        self * self
116    }
117
118    fn inverse(&self) -> Self;
119    fn pow2523(&self) -> Self;
120}
