13/// A X25519 secret key.
14///
15/// This is a wrapper around a `Scalar`. To obtain the corresponding `PublicKey`,
16/// use `PublicKey::from(&secret_key)`.
17#[derive(Clone)]
18pub struct SecretKey(pub(crate) Scalar);
19
20/// The result of a Diffie-Hellman key exchange.
21///
22/// Each party computes this using their [`SecretKey`] and their counterparty's [`PublicKey`].
23#[derive(Clone, Zeroize, ZeroizeOnDrop)]
24pub struct SharedSecret(pub(crate) MontgomeryPoint);
25
26impl From<[u8; 32]> for PublicKey {
27    /// Given a byte array, construct a x25519 `PublicKey`.
28    fn from(bytes: [u8; 32]) -> Self {
29        let field_element = FieldElement::from_unreduced_bytes(&bytes);
30        PublicKey(MontgomeryPoint(field_element))
31    }
32}
33
34impl<'a> From<&'a SecretKey> for PublicKey {
35    /// Given an x25519 [`SecretKey`] key, compute its corresponding [`PublicKey`].
36    fn from(secret: &'a SecretKey) -> PublicKey {
37        let public_point = &secret.0 * &MontgomeryPoint::basepoint();
38        PublicKey(public_point)
39    }
40}

63impl SecretKey {
64    /// Perform a Diffie-Hellman key agreement between `self` and
65    /// `their_public` key to produce a `SharedSecret`.
66    pub fn agree(&self, their_public: &PublicKey) -> SharedSecret {
67        SharedSecret(&self.0 * &their_public.0)
68    }
69
70    pub fn from_seed(seed: &[u8; SECRETKEY_SEED_LENGTH]) -> Self {
71        Self(clamp_scalar(*seed))
72    }
73
74    /// Extract this key's bytes for serialization.
75    pub fn to_bytes(&self) -> [u8; 32] {
76        self.0.to_bytes()
77    }
78
79    /// Corresponding public key.
80    pub fn public(&self) -> PublicKey {
81        self.into()
82    }
83}

85/// "Decode" a scalar from a 32-byte array.
86///
87/// By "decode" here, what is really meant is applying key clamping by twiddling
88/// some bits.
89///
90/// # Returns
91///
92/// A `Scalar`.
93fn clamp_scalar(mut scalar: [u8; 32]) -> Scalar {
94    scalar[0] &= 248;
95    scalar[31] &= 127;
96    scalar[31] |= 64;
97
98    // this is actually redundant, it does `s.0[31] &= 0b0111_1111;`,
99    // i.e., set highest bit to zero, enforcing s < 2^255
100    // scalar_from_bits(scalar)
101
102    Scalar(scalar)
103}
104
105/// Implementations:
106/// - MUST mask highest bit in input_u
107/// - MUST accept non-canonical input_u, reduce modulo base field
108///
109/// <https://tools.ietf.org/html/rfc7748#section-5>
110pub fn x25519(scalar: [u8; 32], input_u: [u8; 32]) -> [u8; 32] {
111    let scalar = clamp_scalar(scalar);
112    let secret_key = SecretKey(scalar);
113
114    let public_key = PublicKey::from(input_u);
115
116    let agreed_secret = secret_key.agree(&public_key);
117
118    agreed_secret.0.to_bytes()
119}

228    #[test]
229    fn zeroize_on_drop() {
230        let mut secret = SecretKey::from_seed(&[1u8; 32]);
231        let public = PublicKey::from([2u8; 32]);
232        let mut shared_secret = secret.agree(&public);
233
234        assert_ne!(secret.0.as_bytes(), &[0u8; 32]);
235
236        unsafe {
237            core::ptr::drop_in_place(&mut secret);
238        }
239
240        assert_eq!(secret.0.as_bytes(), &[0u8; 32]);
241
242        assert_ne!(shared_secret.0.to_bytes(), [0u8; 32]);
243
244        unsafe {
245            core::ptr::drop_in_place(&mut shared_secret);
246        }
247
248        assert_eq!(shared_secret.0.to_bytes(), [0u8; 32]);
249    }
