23/// Holds the \\(u\\)-coordinate of a point on the Montgomery form of
24/// Curve25519 or its twist.
25#[derive(Clone, Copy, Debug, Default, Zeroize)]

43impl MontgomeryPoint {
44    // /// View this `MontgomeryPoint` as an array of bytes.
45    // pub fn as_bytes(&self) -> &[u8; 32] {
46    //     &self.0.to_bytes()
47    // }
48
49    /// Convert this `MontgomeryPoint` to an array of bytes.
50    pub fn to_bytes(&self) -> [u8; 32] {
51        self.0.to_bytes()
52    }
53
54    /// Attempt to convert to an `EdwardsPoint`, using the supplied
55    /// choice of sign for the `EdwardsPoint`.
56    ///
57    /// # Inputs
58    ///
59    /// * `sign`: a `u8` donating the desired sign of the resulting
60    ///   `EdwardsPoint`.  `0` denotes positive and `1` negative.
61    ///
62    /// # Return
63    ///
64    /// * `Some(EdwardsPoint)` if `self` is the \\(u\\)-coordinate of a
65    /// point on (the Montgomery form of) Curve25519;
66    ///
67    /// * `None` if `self` is the \\(u\\)-coordinate of a point on the
68    /// twist of (the Montgomery form of) Curve25519;
69    ///
70    pub fn to_edwards(&self, sign: u8) -> Result<EdwardsPoint> {
71        // To decompress the Montgomery u coordinate to an
72        // `EdwardsPoint`, we apply the birational map to obtain the
73        // Edwards y coordinate, then do Edwards decompression.
74        //
75        // The birational map is y = (u-1)/(u+1).
76        //
77        // The exceptional points are the zeros of the denominator,
78        // i.e., u = -1.
79        //
80        // But when u = -1, v^2 = u*(u^2+486662*u+1) = 486660.
81        //
82        // Since this is nonsquare mod p, u = -1 corresponds to a point
83        // on the twist, not the curve, so we can reject it early.
84
85        let u = &self.0;
86        let one = &FieldElement::ONE;
87
88        if u + one == FieldElement::ZERO {
89            return Err(Error::WrongTwist);
90        }
91
92        let y = &(u - one) * &(u + one).inverse();
93
94        let mut y_bytes = y.to_bytes();
95        y_bytes[31] ^= sign << 7;
96
97        CompressedEdwardsY(y_bytes).decompressed()
98    }
99
100    pub fn basepoint() -> Self {
101        Self(FieldElement::MONTGOMERY_BASEPOINT_U)
102    }
103}

105/// A `ProjectivePoint` holds a point on the projective line
106/// \\( \mathbb P(\mathbb F\_p) \\), which we identify with the Kummer
107/// line of the Montgomery curve.
108#[derive(Copy, Clone, Debug)]
109#[allow(non_snake_case)]
110struct ProjectivePoint {
111    pub U: FieldElement,
112    pub W: FieldElement,
113}

116impl ProjectivePoint {

118        ProjectivePoint {
119            U: FieldElement::ONE,
120            W: FieldElement::ZERO,
121        }
122    }
123}

131impl ConditionallySelectable for ProjectivePoint {
132    fn conditional_select(
133        a: &ProjectivePoint,
134        b: &ProjectivePoint,
135        choice: Choice,
136    ) -> ProjectivePoint {
137        ProjectivePoint {
138            U: FieldElement::conditional_select(&a.U, &b.U, choice),
139            W: FieldElement::conditional_select(&a.W, &b.W, choice),
140        }
141    }
142}

144impl ProjectivePoint {
145    /// Dehomogenize this point to affine coordinates.
146    ///
147    /// # Return
148    ///
149    /// * \\( u = U / W \\) if \\( W \neq 0 \\);
150    /// * \\( 0 \\) if \\( W \eq 0 \\);
151    pub fn to_affine(self) -> MontgomeryPoint {
152        let u = &self.U * &self.W.inverse();
153        MontgomeryPoint(u)
154    }

157/// Perform the double-and-add step of the Montgomery ladder.
158///
159/// Given projective points
160/// \\( (U\_P : W\_P) = u(P) \\),
161/// \\( (U\_Q : W\_Q) = u(Q) \\),
162/// and the affine difference
163/// \\(      u\_{P-Q} = u(P-Q) \\), set
164/// $$
165///     (U\_P : W\_P) \gets u([2]P)
166/// $$
167/// and
168/// $$
169///     (U\_Q : W\_Q) \gets u(P + Q).
170/// $$
171#[allow(non_snake_case)]
172fn differential_add_and_double(
173    P: &mut ProjectivePoint,
174    Q: &mut ProjectivePoint,
175    affine_PmQ: &FieldElement,
176) {
177    let t0 = &P.U + &P.W;
178    let t1 = &P.U - &P.W;
179    let t2 = &Q.U + &Q.W;
180    let t3 = &Q.U - &Q.W;
181
182    let t4 = t0.squared(); // (U_P + W_P)^2 = U_P^2 + 2 U_P W_P + W_P^2
183    let t5 = t1.squared(); // (U_P - W_P)^2 = U_P^2 - 2 U_P W_P + W_P^2
184
185    let t6 = &t4 - &t5; // 4 U_P W_P
186
187    let t7 = &t0 * &t3; // (U_P + W_P) (U_Q - W_Q) = U_P U_Q + W_P U_Q - U_P W_Q - W_P W_Q
188    let t8 = &t1 * &t2; // (U_P - W_P) (U_Q + W_Q) = U_P U_Q - W_P U_Q + U_P W_Q - W_P W_Q
189
190    let t9 = &t7 + &t8; // 2 (U_P U_Q - W_P W_Q)
191    let t10 = &t7 - &t8; // 2 (W_P U_Q - U_P W_Q)
192
193    let t11 = t9.squared(); // 4 (U_P U_Q - W_P W_Q)^2
194    let t12 = t10.squared(); // 4 (W_P U_Q - U_P W_Q)^2
195
196    let t13 = &FieldElement::APLUS2_OVER_FOUR * &t6; // (A + 2) U_P U_Q
197
198    let t14 = &t4 * &t5; // ((U_P + W_P)(U_P - W_P))^2 = (U_P^2 - W_P^2)^2
199    let t15 = &t13 + &t5; // (U_P - W_P)^2 + (A + 2) U_P W_P
200
201    let t16 = &t6 * &t15; // 4 (U_P W_P) ((U_P - W_P)^2 + (A + 2) U_P W_P)
202
203    let t17 = affine_PmQ * &t12; // U_D * 4 (W_P U_Q - U_P W_Q)^2
204    let t18 = t11; // W_D * 4 (U_P U_Q - W_P W_Q)^2
205
206    P.U = t14; // U_{P'} = (U_P + W_P)^2 (U_P - W_P)^2
207    P.W = t16; // W_{P'} = (4 U_P W_P) ((U_P - W_P)^2 + ((A + 2)/4) 4 U_P W_P)
208    Q.U = t18; // U_{Q'} = W_D * 4 (U_P U_Q - W_P W_Q)^2
209    Q.W = t17; // W_{Q'} = U_D * 4 (W_P U_Q - U_P W_Q)^2
210}

212/// Multiply this `MontgomeryPoint` by a `Scalar`.
213impl<'a, 'b> Mul<&'b Scalar> for &'a MontgomeryPoint {
214    type Output = MontgomeryPoint;
215
216    /// Given `self` \\( = u\_0(P) \\), and a `Scalar` \\(n\\), return \\( u\_0([n]P) \\).
217    fn mul(self, scalar: &'b Scalar) -> MontgomeryPoint {
218        // Algorithm 8 of Costello-Smith 2017
219        let affine_u = self.0;
220        let mut x0 = ProjectivePoint::neutral_element();
221        let mut x1 = ProjectivePoint {
222            U: affine_u,
223            W: FieldElement::ONE,
224        };
225
226        let bits: [i8; 256] = scalar.bits();
227
228        for i in (0..255).rev() {
229            let choice: u8 = (bits[i + 1] ^ bits[i]) as u8;
230            debug_assert!(choice == 0 || choice == 1);
231
232            ProjectivePoint::conditional_swap(&mut x0, &mut x1, choice.into());
233            differential_add_and_double(&mut x0, &mut x1, &affine_u);
234        }
235        ProjectivePoint::conditional_swap(&mut x0, &mut x1, Choice::from(bits[0] as u8));
236
237        x0.to_affine()
238    }
