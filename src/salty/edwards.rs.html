23/// "Compressed" form of a `EdwardsPoint`, whereby
24/// the sign of the x-coordinate is stuffed in a
25/// spare bit of the y-coordinate
26#[derive(Clone, Copy, Debug, Default)]

35impl CompressedY {
36    /// This is rather tricky: to get the x-coordinate,
37    /// and not just its sign, need to calculate the square
38    /// root of `u/v := (y**2 - 1)/(dy**2 + 1)`. Moreover, we want
39    /// to detect whether our compressed Y actually corresponds
40    /// to a point on the curve! The original sources are
41    /// [the Tweet NaCl paper, section 5](tweetnacl) and
42    /// the [ed25519 paper][ed25519], also section 5.
43    ///
44    /// [tweetnacl]: http://tweetnacl.cr.yp.to/tweetnacl-20140917.pdf
45    /// [ed25519]: https://cryptojedi.org/papers/ed25519-20110926.pdf
46    pub fn decompressed(&self) -> Result<EdwardsPoint> {
47        #![allow(non_snake_case)]
48
49        // point = (X, Y, Z, T)
50        //
51        // basic strategy: use exponentiation by `2**252 - 3`,
52        // which "has all bits set except position 1".
53
54        // TODO: actually implement TryFrom
55        // let Y = FieldElement::try_from(self.as_bytes())?;
56        let Y = FieldElement::from_bytes_unchecked(self.as_bytes());
57        let Z = FieldElement::ONE;
58        let Y_squared = Y.squared();
59
60        let u = &Y_squared - &Z; // aka num[erator], y**2 - 1
61        let v = &(&Y_squared * &FieldElement::D) + &Z; // aka den[ominator], dy**2 + 1
62
63        let v2 = v.squared();
64        let v4 = v2.squared();
65        let v7 = &(&v4 * &v2) * &v;
66
67        let t = &v7 * &u; // term: t = uv**7
68        let mut X = &(&(&t.pow2523() * &u) * &v2) * &v; // aka `beta`
69
70        let chk = &X.squared() * &v;
71        if chk != u {
72            X = &X * &FieldElement::I;
73        }
74
75        let chk = &X.squared() * &v;
76        if chk != u {
77            return Err(Error::PublicKeyBytesInvalid);
78        }
79
80        // I really don't get it. TweetNaCl checks for equality.
81        // If we do that, our tests fail. This way, tests pass.
82        if X.parity() != (self.0[31] >> 7) {
83            X = -&X;
84        }
85
86        let T = &X * &Y;
87        Ok(EdwardsPoint([X, Y, Z, T]))
88    }

93    //   unpack25519(r[1],p); // Y = compressed Y with x's sign bit erased
94    //   S(num,r[1]);
95    //   M(den,num,D);
96    //   Z(num,num,r[2]);
97    //   A(den,r[2],den); // set numerator, denominator as above
98
99    //   S(den2,den);
100    //   S(den4,den2);
101    //   M(den6,den4,den2);
102    //   M(t,den6,num);
103    //   M(t,t,den);  // set t = denominator**7 * numerator
104
105    //   pow2523(t,t);
106    //   M(t,t,num);
107    //   M(t,t,den);
108    //   M(t,t,den);
109    //   M(r[0],t,den); // X = sqrt(t)*num*den**3

111    //   S(chk,r[0]);

113    //   if (neq25519(chk, num)) M(r[0],r[0],I);

126impl EdwardsPoint {
127    pub fn basepoint() -> EdwardsPoint {
128        EdwardsPoint([
129            FieldElement::EDWARDS_BASEPOINT_X,
130            FieldElement::EDWARDS_BASEPOINT_Y,
131            FieldElement::ONE,
132            &FieldElement::EDWARDS_BASEPOINT_X * &FieldElement::EDWARDS_BASEPOINT_Y,
133        ])
134    }
135
136    pub fn neutral_element() -> EdwardsPoint {
137        EdwardsPoint([
138            FieldElement::ZERO,
139            FieldElement::ONE,
140            FieldElement::ONE,
141            FieldElement::ZERO,
142        ])
143    }
144
145    pub fn compressed(&self) -> CompressedY {
146        // normalize X, Y to Z = 1
147        let z_inverse = &self.0[2].inverse();
148        let x = &self.0[0] * z_inverse;
149        let y = &self.0[1] * z_inverse;
150
151        // normalized Y coordinate
152        let mut r = y.to_bytes();
153        // slot sign of X in the "spare" top bit of last byte
154        // dalek calls this parity "is_negative"
155        r[31] ^= x.parity() << 7;
156
157        CompressedY(r)
158    }
159
160    /// Convert this `EdwardsPoint` on the Edwards model to the
161    /// corresponding `MontgomeryPoint` on the Montgomery model.
162    ///
163    /// This function has one exceptional case; the identity point of
164    /// the Edwards curve is sent to the 2-torsion point \\((0,0)\\)
165    /// on the Montgomery curve.
166    ///
167    /// Note that this is a one-way conversion, since the Montgomery
168    /// model does not retain sign information.
169    pub fn to_montgomery(&self) -> MontgomeryPoint {
170        //// We have u = (1+y)/(1-y) = (Z+Y)/(Z-Y).
171        ////
172        //// The denominator is zero only when y=1, the identity point of
173        //// the Edwards curve.  Since 0.invert() = 0, in this case we
174        //// compute the 2-torsion point (0,0).
175        //let U = &self.Z + &self.Y;
176        //let W = &self.Z - &self.Y;
177        //let u = &U * &W.invert();
178        //MontgomeryPoint(u.to_bytes())
179        MontgomeryPoint(self.u())
180    }
181
182    /// The x-coordinate of the point
183    pub fn x(&self) -> FieldElement {
184        let z_inverse = &self.0[2].inverse();
185
186        &self.0[0] * z_inverse
187    }
188
189    /// The y-coordinate of the point
190    pub fn y(&self) -> FieldElement {
191        let z_inverse = &self.0[2].inverse();
192
193        &self.0[1] * z_inverse
194    }
195
196    /// The u-coordinate of the X25519 point
197    pub fn u(&self) -> FieldElement {
198        let y = self.y();
199        let one = FieldElement::ONE;
200
201        &(&y + &one) * &(&one - &y).inverse()
202    }
203}

205impl<'a, 'b> Add<&'b EdwardsPoint> for &'a EdwardsPoint {
206    type Output = EdwardsPoint;
207
208    fn add(self, other: &'b EdwardsPoint) -> Self::Output {
209        let p = &self.0;
210        let q = &other.0;
211
212        let a = &p[1] - &p[0];
213        let t = &q[1] - &q[0];
214        let a = &a * &t; // A <- (Y1 - X1)(Y2 - X2)
215
216        // let mut b = &p[0] + &p[1];
217        let b = &p[0] + &p[1];
218        let t = &q[0] + &q[1];
219        let b = &b * &t; // B <- (Y1 + X1)*(Y2 + X2)
220                         // b *= &t;
221
222        let c = &p[3] * &q[3];
223        let c = &c * &FieldElement::D2; // C <- k*T1*T2  with k = 2d' =
224
225        let d = &p[2] * &q[2];
226        let d = &d + &d; // D <- 2*Z1*Z2
227
228        let e = &b - &a;
229        let f = &d - &c;
230        let g = &d + &c;
231        let h = &b + &a;
232
233        let coordinates = [&e * &f, &h * &g, &g * &f, &e * &h];
234
235        EdwardsPoint(coordinates)
236    }
237}

248impl<'a, 'b> Mul<&'b EdwardsPoint> for &'a Scalar {
249    type Output = EdwardsPoint;
250
251    fn mul(self, point: &'b EdwardsPoint) -> EdwardsPoint {
252        let mut p = EdwardsPoint([
253            FieldElement::ZERO,
254            FieldElement::ONE,
255            FieldElement::ONE,
256            FieldElement::ZERO,
257        ]);
258
259        let mut q = *point;
260        let scalar = self;
261
262        for i in (0..=255).rev() {
263            let b = ((scalar.0[i / 8] >> (i & 7)) & 1).into();
264            EdwardsPoint::conditional_swap(&mut p, &mut q, b);
265
266            q = &q + &p;
267            p = &p + &p;
268
269            EdwardsPoint::conditional_swap(&mut p, &mut q, b);
270        }
271
272        p
273    }
274}

276impl ConditionallySelectable for EdwardsPoint {
277    fn conditional_select(p: &Self, q: &Self, choice: Choice) -> Self {
278        let mut selection = Self::default();
279
280        for (i, (pi, qi)) in p.0.iter().zip(q.0.iter()).enumerate() {
281            selection.0[i] = FieldElement::conditional_select(pi, qi, choice);
282        }
283        selection
284    }
285
286    fn conditional_swap(p: &mut Self, q: &mut Self, choice: Choice) {
287        for (pi, qi) in p.0.iter_mut().zip(q.0.iter_mut()) {
288            FieldElement::conditional_swap(pi, qi, choice);
289        }
290    }
291}
