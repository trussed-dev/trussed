8// this is `rotate-right(x, n)` for 64-bit words
9// implicitly, 0 <= n < 64
10fn R(w: Wrapping<u64>, n: usize) -> Wrapping<u64> {
11    (w >> n) | (w << (64 - n))
12}
13#[allow(non_snake_case)]
14// this is "choose", input `x` picks the output bit from y or z:
15// if bit `i` of `x` is 1, then output is bit `i` from `y`,
16// else bit `i` from `z`.
17fn Ch(x: Wrapping<u64>, y: Wrapping<u64>, z: Wrapping<u64>) -> Wrapping<u64> {
18    (x & y) ^ (!x & z)
19}
20#[allow(non_snake_case)]
21// this is "majority", each bit is the majority of the
22// three input bits of x, y, z at this index
23fn Maj(x: Wrapping<u64>, y: Wrapping<u64>, z: Wrapping<u64>) -> Wrapping<u64> {
24    (x & y) ^ (x & z) ^ (y & z)
25}
26#[allow(non_snake_case)]
27fn Sigma0(x: Wrapping<u64>) -> Wrapping<u64> {
28    R(x, 28) ^ R(x, 34) ^ R(x, 39)
29}
30#[allow(non_snake_case)]
31fn Sigma1(x: Wrapping<u64>) -> Wrapping<u64> {
32    R(x, 14) ^ R(x, 18) ^ R(x, 41)
33}
34fn sigma0(x: Wrapping<u64>) -> Wrapping<u64> {
35    R(x, 1) ^ R(x, 8) ^ (x >> 7)
36}
37fn sigma1(x: Wrapping<u64>) -> Wrapping<u64> {
38    R(x, 19) ^ R(x, 61) ^ (x >> 6)
39}

67fn hash_blocks(digest: &mut [u8; 64], msg: &[u8]) -> usize {
68    #![allow(non_snake_case)]
69
70    // convert digest (u8-array) into hash parts (u64-words array)
71    let mut H: [Wrapping<u64>; 8] = Default::default(); //[Wrapping(0); 8];
72    for (h, chunk) in H.iter_mut().zip(digest.chunks(8)) {
73        // *h = Wrapping(BigEndian::read_u64(chunk));
74        *h = Wrapping(u64::from_be_bytes(chunk.try_into().unwrap()));
75    }
76
77    let unprocessed = msg.len() & 127; // remainder modulo 128
78    for block in msg[..msg.len() - unprocessed].chunks(128) {
79        // W is the "message schedule", it is updated below
80        // This is like Section 6.1.3 from FIPS 180.
81        let mut W: [Wrapping<u64>; 16] = Default::default(); //[Wrapping(0); 16];
82        for (w, chunk) in W.iter_mut().zip(block.chunks(8)) {
83            *w = Wrapping(u64::from_be_bytes(chunk.try_into().unwrap()));
84        }
85
86        // initialize "working variables" with previous hash
87        let mut a = H[0];
88        let mut b = H[1];
89        let mut c = H[2];
90        let mut d = H[3];
91        let mut e = H[4];
92        let mut f = H[5];
93        let mut g = H[6];
94        let mut h = H[7];
95
96        // apply 80 rounds
97        for t in 0..80 {
98            let T1 = h + Sigma1(e) + Ch(e, f, g) + Wrapping(K[t]) + W[t % 16];
99            let T2 = Sigma0(a) + Maj(a, b, c);
100            h = g;
101            g = f;
102            f = e;
103            e = d + T1;
104            d = c;
105            c = b;
106            b = a;
107            a = T1 + T2;
108
109            // update message schedule if necessary
110            if t % 16 == 15 {
111                // TODO: need Wprev??
112                // let Wprev = W.clone();
113                // for j in 0..16 {
114                //     W[j] +=      Wprev[(j +  9) % 16]
115                //         + sigma0(Wprev[(j +  1) % 16])
116                //         + sigma1(Wprev[(j + 14) % 16]);
117                // }
118                for j in 0..16 {
119                    W[j] += W[(j + 9) % 16] + sigma0(W[(j + 1) % 16]) + sigma1(W[(j + 14) % 16]);
120                }
121            }
122        }
123        H[0] += a;
124        H[1] += b;
125        H[2] += c;
126        H[3] += d;
127        H[4] += e;
128        H[5] += f;
129        H[6] += g;
130        H[7] += h;
131    }
132
133    // convert hash parts (u64-words array) back into digest (u8-array)
134    for (d, h) in digest.chunks_mut(8).zip(H.iter()) {
135        // BigEndian::write_u64(d, h.0);
136        d.copy_from_slice(&h.0.to_be_bytes());
137    }
138
139    unprocessed
140}

156// generates a 64 bytes hash of the `msg`
157// https://dx.doi.org/10.6028/NIST.FIPS.180-4
158//
159// steps:
160// - pad message to obtain 128 byte blocks
161// - start with IV
162// - "hash in" each block in turn
163#[allow(dead_code)]
164pub fn sha512(digest: &mut [u8; 64], msg: &[u8]) {
165    let l = msg.len();
166    // assert!(l >> 125 == 0);  // u128-encoded length is in bits
167
168    // initialize digest with the initialisation vector
169    // let mut h: [u8; 64] = IV.clone();
170    digest.copy_from_slice(&IV);
171
172    // hash full (128 bytes) blocks from message
173    let unprocessed = hash_blocks(digest, msg);
174
175    // generate padding (can be 1 or 2 blocks of 128 bytes)
176    let mut padding: [u8; 256] = [0u8; 256];
177    let padding_length = match unprocessed < 112 {
178        true => 128,
179        false => 256,
180    };
181    // first: remaining message
182    padding[..unprocessed].copy_from_slice(&msg[l - unprocessed..]);
183    // then: bit 1 followed by zero bits until...
184    padding[unprocessed] = 128;
185    // ...message length in bits (NB: l is in bytes)
186    #[cfg(target_pointer_width = "64")]
187    {
188        padding[padding_length - 9] = (l >> 61) as u8;
189    }
190    // BigEndian::write_u64(&mut padding[padding_length - 8..], (l << 3) as u64);
191    padding[padding_length - 8..].copy_from_slice(&((l << 3) as u64).to_be_bytes());
192
193    let padding = &padding[..padding_length];
194    hash_blocks(digest, padding);
195
196    // digest.copy_from_slice(&h);
197}

225    pub fn update(&mut self, data: &[u8]) {
226        self.data_length += data.len();
227
228        // if self.unprocessed + data.len() < 128 {
229        if (self.unprocessed + data.len()) & !(0x80 - 1) == 0 {
230            self.buffer[self.unprocessed..self.unprocessed + data.len()].copy_from_slice(data);
231            self.unprocessed += data.len();
232        } else {
233            // fill up buffer
234            let filler = 128 - self.unprocessed;
235            self.buffer[self.unprocessed..].copy_from_slice(&data[..filler]);
236            hash_blocks(&mut self.digest, &self.buffer);
237
238            self.unprocessed = hash_blocks(&mut self.digest, &data[filler..]);
239            self.buffer[..self.unprocessed].copy_from_slice(&data[data.len() - self.unprocessed..]);
240        }
241    }

243    pub fn updated(mut self, data: &[u8]) -> Self {
244        self.update(data);
