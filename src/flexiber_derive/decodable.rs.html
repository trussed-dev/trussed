1use proc_macro2::TokenStream;
2use quote::{quote, ToTokens};
3use syn::{Attribute, DataStruct, Field, Ident};
4use synstructure::Structure;

6use crate::{extract_attrs_optional_tag, FieldAttrs, Tag};

8/// Derive Decodable on a struct
9pub(crate) struct DeriveDecodableStruct {
10    /// Field decoders
11    decode_fields: TokenStream,
12
13    /// Bound fields of a struct to be returned
14    decode_result: TokenStream,
15}
16
17impl DeriveDecodableStruct {
18    pub fn derive(s: Structure<'_>, data: &DataStruct, name: &Ident, attrs: &[Attribute]) -> TokenStream {
19
20        let (tag, _slice) = extract_attrs_optional_tag(name, attrs);
21
22        let mut state = Self {
23            decode_fields: TokenStream::new(),
24            decode_result: TokenStream::new(),
25        };
26
27        for field in &data.fields {
28            state.derive_field(field);
29        }
30
31        state.finish(&s, tag)
32    }
33
34    /// Derive handling for a particular `#[field(...)]`
35    fn derive_field(&mut self, field: &Field) {
36        let attrs = FieldAttrs::new(field);
37        self.derive_field_decoder(&attrs);
38    }
39
40    /// Derive code for decoding a field of a message
41    fn derive_field_decoder(&mut self, field: &FieldAttrs) {
42        let field_name = &field.name;
43        let tag = field.tag;
44
45        let field_decoder = match tag {
46            Tag::Ber(tag) => {
47                let class = tag.class as u8;
48                let constructed = tag.constructed;
49                let tag_number = tag.number;
50
51                if field.slice {
52                    quote! {
53                        let tag = ::flexiber::Tag::from(
54                            flexiber::Class::try_from(#class).unwrap(),
55                            #constructed,
56                            #tag_number
57                        );
58                        let #field_name =
59                            decoder.decode_tagged_slice(tag)?.try_into().unwrap();
60                                // .map_err(|_| flexiber::ErrorKind::Length { tag })?;
61                    }
62                } else {
63                    quote! {
64                        let tag = ::flexiber::Tag::from(
65                            flexiber::Class::try_from(#class).unwrap(),
66                            #constructed,
67                            #tag_number
68                        );
69
70                        let #field_name = decoder.decode_tagged_value(tag)?;
71                    }
72                }
73            }
74            Tag::Simple(tag) => {
75                let field_tag = tag.0;
76                if field.slice {
77                    quote! { let #field_name =
78                        decoder.decode_tagged_slice(::flexiber::SimpleTag::try_from(#field_tag).unwrap())?.try_into()
79                            .map_err(|_| {
80                                use flexiber::TagLike;
81                                flexiber::ErrorKind::Length { tag: flexiber::SimpleTag::try_from(#field_tag).unwrap().embedding() }
82                            })?;
83                    }
84                } else {
85                    quote! { let #field_name = decoder.decode_tagged_value(::flexiber::SimpleTag::try_from(#field_tag).unwrap())?; }
86                }
87            }
88        };
89        field_decoder.to_tokens(&mut self.decode_fields);
90
91        let field_result = quote!(#field_name,);
92        field_result.to_tokens(&mut self.decode_result);
93    }
94
95    /// Finish deriving a struct
96    fn finish(self, s: &Structure<'_>, tag: Option<Tag>) -> TokenStream {
97
98        let decode_fields = self.decode_fields;
99        let decode_result = self.decode_result;
100
101        if let Some(tag) = tag {
102
103            match tag {
104                Tag::Ber(tag) => {
105                    let class = tag.class as u8;
106                    let constructed = tag.constructed;
107                    let tag_number = tag.number;
108
109                    s.gen_impl(quote! {
110                        gen impl<'a> core::convert::TryFrom<flexiber::TaggedSlice<'a>> for @Self {
111                            type Error = flexiber::Error;
112
113                            fn try_from(tagged_slice: flexiber::TaggedSlice<'a>) -> flexiber::Result<Self> {
114                                use core::convert::TryInto;
115                                use flexiber::TagLike;
116                                let tag = ::flexiber::Tag::from(
117                                    flexiber::Class::try_from(#class).unwrap(),
118                                    #constructed,
119                                    #tag_number
120                                );
121                                tagged_slice.tag().assert_eq(tag)?;
122                                tagged_slice.decode_nested(|decoder| {
123                                    #decode_fields
124
125                                    Ok(Self { #decode_result })
126                                })
127                            }
128                        }
129                    })
130                }
131                Tag::Simple(tag) => {
132                    let tag = tag.0;
133                    s.gen_impl(quote! {
134                        gen impl<'a> Decodable<'a> for @Self {
135                            fn decode(decoder: &mut Decoder<'a>) -> Result<Self> {
136                                flexiber::TaggedSlice::<'a, flexiber::SimpleTag>::decode(decoder)
137                                    .and_then(|tagged_slice| {
138                                        use core::convert::TryInto;
139                                        use flexiber::TagLike;
140                                        tagged_slice.tag().assert_eq(flexiber::SimpleTag::try_from(#tag).unwrap())?;
141                                        tagged_slice.decode_nested(|decoder| {
142                                            #decode_fields
143
144                                            Ok(Self { #decode_result })
145                                        })
146                                    })
147                                    .or_else(|e| decoder.error(e.kind()))
148                            }
149
150                        }
151                    })
152                }
153            }
154        } else {
155            s.gen_impl(quote! {
156                gen impl<'a> flexiber::Decodable<'a> for @Self {
157                    fn decode(decoder: &mut flexiber::Decoder<'a>) -> flexiber::Result<Self> {
158                        use core::convert::{TryFrom, TryInto};
159                        #decode_fields
160                        Ok(Self { #decode_result })
161                    }
162                }
163            })
164        }
165    }
166}
