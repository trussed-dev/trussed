67/// Object-safe trait for files.
68///
69/// The methods for opening files cannot be implemented in this trait.  Use these methods instead:
70/// - [`DynFilesystem::create_file_and_then`](trait.DynFilesystem.html#method.create_file_and_then)
71/// - [`DynFilesystem::open_file_and_then`](trait.DynFilesystem.html#method.open_file_and_then)
72/// - [`DynFilesystem::open_file_with_options_and_then`](trait.DynFilesystem.html#method.open_file_with_options_and_then)
73pub trait DynFile: Read + Seek + Write {
74    fn sync(&self) -> Result<()>;
75    fn len(&self) -> Result<usize>;
76    fn is_empty(&self) -> Result<bool>;
77    fn set_len(&self, size: usize) -> Result<()>;
78}
79
80impl dyn DynFile + '_ {
81    pub fn read_to_end<V: Vec>(&self, buf: &mut V) -> Result<usize> {
82        let had = buf.as_ref().len();
83        buf.resize_to_capacity();
84        let read = self.read(&mut buf.as_mut()[had..])?;
85        buf.truncate(had + read);
86        Ok(read)
87    }
88}

90/// Object-safe trait for filesystems.
91///
92/// The following methods cannot support generic return types in the callbacks:
93/// - [`DynFilesystem::create_file_and_then_unit`][]
94/// - [`DynFilesystem::open_file_and_then_unit`][]
95/// - [`DynFilesystem::open_file_with_flags_and_then_unit`][]
96/// - [`DynFilesystem::read_dir_and_then_unit`][]
97///
98/// Use these helper functions instead:
99/// - [`DynFilesystem::create_file_and_then`](#method.create_file_and_then)
100/// - [`DynFilesystem::open_file_and_then`](#method.open_file_and_then)
101/// - [`DynFilesystem::open_file_with_flags_and_then`](#method.open_file_with_flags_and_then)
102/// - [`DynFilesystem::read_dir_and_then`](#method.read_dir_and_then)
103pub trait DynFilesystem {
104    fn total_blocks(&self) -> usize;
105    fn total_space(&self) -> usize;
106    fn available_blocks(&self) -> Result<usize>;
107    fn available_space(&self) -> Result<usize>;
108    fn remove(&self, path: &Path) -> Result<()>;
109    fn remove_dir(&self, path: &Path) -> Result<()>;
110    fn remove_dir_all(&self, path: &Path) -> Result<()>;
111    fn remove_dir_all_where(&self, path: &Path, predicate: Predicate<'_>) -> Result<usize>;
112    fn rename(&self, from: &Path, to: &Path) -> Result<()>;
113    fn exists(&self, path: &Path) -> bool;
114    fn metadata(&self, path: &Path) -> Result<Metadata>;
115    fn create_file_and_then_unit(&self, path: &Path, f: FileCallback<'_>) -> Result<()>;
116    fn open_file_and_then_unit(&self, path: &Path, f: FileCallback<'_>) -> Result<()>;
117    fn open_file_with_flags_and_then_unit(
118        &self,
119        flags: FileOpenFlags,
120        path: &Path,
121        f: FileCallback<'_>,
122    ) -> Result<()>;
123    fn attribute<'a>(
124        &self,
125        path: &Path,
126        id: u8,
127        buffer: &'a mut [u8],
128    ) -> Result<Option<Attribute<'a>>>;
129    fn remove_attribute(&self, path: &Path, id: u8) -> Result<()>;
130    fn set_attribute(&self, path: &Path, id: u8, data: &[u8]) -> Result<()>;
131    fn read_dir_and_then_unit(&self, path: &Path, f: DirEntriesCallback<'_>) -> Result<()>;
132    fn create_dir(&self, path: &Path) -> Result<()>;
133    fn create_dir_all(&self, path: &Path) -> Result<()>;
134    fn write(&self, path: &Path, contents: &[u8]) -> Result<()>;
135    fn write_chunk(&self, path: &Path, contents: &[u8], pos: OpenSeekFrom) -> Result<()>;
136}

138impl dyn DynFilesystem + '_ {
139    pub fn read<V: Vec>(&self, path: &Path) -> Result<V> {
140        let mut contents = V::default();
141        self.open_file_and_then(path, &mut |file| {
142            file.read_to_end(&mut contents)?;
143            Ok(())
144        })?;
145        Ok(contents)
146    }
147
148    pub fn read_chunk<V: Vec>(&self, path: &Path, pos: OpenSeekFrom) -> Result<(V, usize)> {
149        let mut contents = V::default();
150        contents.resize_to_capacity();
151        let file_len = self.open_file_and_then(path, &mut |file| {
152            file.seek(pos.into())?;
153            let read_n = file.read(contents.as_mut())?;
154            contents.truncate(read_n);
155            file.len()
156        })?;
157        Ok((contents, file_len))
158    }
159
160    pub fn create_file_and_then<R>(&self, path: &Path, f: FileCallback<'_, R>) -> Result<R> {
161        let mut result = Err(Error::IO);
162        self.create_file_and_then_unit(path, &mut |file| {
163            result = Ok(f(file)?);
164            Ok(())
165        })?;
166        result
167    }
168
169    pub fn open_file_and_then<R>(&self, path: &Path, f: FileCallback<'_, R>) -> Result<R> {
170        let mut result = Err(Error::IO);
171        self.open_file_and_then_unit(path, &mut |file| {
172            result = Ok(f(file)?);
173            Ok(())
174        })?;
175        result
176    }
177
178    pub fn open_file_with_flags_and_then<R>(
179        &self,
180        flags: FileOpenFlags,
181        path: &Path,
182        f: FileCallback<'_, R>,
183    ) -> Result<R> {
184        let mut result = Err(Error::IO);
185        self.open_file_with_flags_and_then_unit(flags, path, &mut |file| {
186            result = Ok(f(file)?);
187            Ok(())
188        })?;
189        result
190    }
191
192    pub fn read_dir_and_then<R>(&self, path: &Path, f: DirEntriesCallback<'_, R>) -> Result<R> {
193        let mut result = Err(Error::IO);
194        self.read_dir_and_then_unit(path, &mut |entries| {
195            result = Ok(f(entries)?);
196            Ok(())
197        })?;
198        result
199    }
