1//! Convenient `Display` and other traits for binary data.
2//!
3//! Standard Rust uses the `fmt::UpperHex` and `LowerHex` traits to implement hexadecimal
4//! representations for use in format strings. For instance,
5//! `format_args!("{:02X?}", &[7, 0xA1, 0xFF])` produces the string `"[07, A1, FF]"`.
6//!
7//! ```
8//! assert_eq!(format!("{:02X?}", [7u8, 0xA1, 0xFF].as_ref()), "[07, A1, FF]");
9//! ```
10//!
11//! However, this only works for the `Debug` trait, not `Display`; needs extra dancing
12//! to pad with leading zeros, and is not very compact when debugging binary data formats.
13//!
14//! The idea of this module is to generate newtypes around byte arrays/slices, and implement the `fmt` traits on these.
15//! In release builds, this is all compiled out and translates to direct instructions for the formatting machinery.
16//!
17//! The ide to implement truncated outputs comes from the `hex_fmt` library, which we refer to
18//! for additional newtypes that can be used.
19//!
20//! ```
21//! use delog::{hex_str, hexstr};
22//!
23//! let data = &[7u8, 0xA1, 255, 0xC7];
24//!
25//! assert_eq!(format!("{}", hexstr!(data)), "07A1FFC7");
26//! assert_eq!(format!("{:x}", hexstr!(data)), "07a1ffc7");
27//! assert_eq!(format!("{:2x}", hex_str!(data)), "07..c7");
28//! assert_eq!(format!("{:<3x}", hex_str!(data)), "07 a1 ff..");
29//! assert_eq!(format!("{:>3x}", hex_str!(data)), "..a1 ff c7");
30//! assert_eq!(format!("{}", hex_str!(data, 2)), "07A1 FFC7");
31//! assert_eq!(format!("{:<3}", hex_str!(data, sep: "|")), "07|A1|FF..");
32//! ```

44/// A type that specifies an unsigned integer.
45///
46/// We use this instead of `typenum` as the latter currently lacks
47/// a mapping from `usize` to the associated type.
48pub trait Unsigned {
49    /// The actual number.
50    const N: usize;
51}
52
53/// Sorry little replacement for the missing int to Unsigned type map in `typenum`.
54#[macro_export]
55#[doc(hidden)]
56macro_rules! typeint {
57    ($name:ident, $n:expr) => {
58        /// A type that represents the integer `N`.
59        pub struct $name;
60        impl $crate::hex::Unsigned for $name {
61            const N: usize = $n;
62        }
63    };
64}

103/// Zero-sized wrapper newtype, allowing grouping bytes in blocks of N hexadecimals
104/// during formatting.
105///
106/// Use the method with the same name to construct this from your byte array or slice,
107/// or preferrably the `hex_str!` or `hexstr!` macro.
108pub struct HexStr<'a, T: ?Sized, U, S>
109where
110    U: Unsigned,
111    S: Separator,
112{
113    /// The value to be formatted.
114    pub value: &'a T,
115    _bytes_per_block: PhantomData<U>,
116    _separator: PhantomData<S>,
117}

119#[macro_export]
120/// Compactly format byte arrays and slices as hexadecimals.
121///
122/// The second parameter refers to the number of bytes in a block (separated by spaces).
123///
124/// ```
125/// use delog::hex_str;
126/// let four_bytes = &[7u8, 0xA1, 255, 0xC7];
127/// assert_eq!(format!("{:x}", hex_str!(four_bytes)), "07 a1 ff c7");
128/// assert_eq!(format!("{}", hex_str!(four_bytes, 2)), "07A1 FFC7");
129/// assert_eq!(format!("{}", hex_str!(four_bytes, 2, sep: "|")), "07A1|FFC7");
130/// assert_eq!(format!("{}", hex_str!(four_bytes, 3)), "07A1FF C7");
131/// ```
132macro_rules! hex_str {
133    ($array:expr) => { $crate::hex_str!($array, 1, sep: " ") };
134    ($array:expr, sep: $separator:expr) => { $crate::hex_str!($array, 1, sep: $separator) };
135    ($array:expr, $n:tt) => { $crate::hex_str!($array, $n, sep: " ") };
136    ($array:expr, $n:tt, sep: $separator:expr) => {{
137        $crate::typesep!(Separator, $separator);
138        $crate::typeint!(Number, $n);
139        $crate::hex::HexStr::<_, Number, Separator>($array)
140    }};
141}

143#[macro_export]
144/// More compactly format byte arrays and slices as hexadecimals.
145///
146/// ```
147/// use delog::hexstr;
148/// let four_bytes = &[7u8, 0xA1, 255, 0xC7];
149/// assert_eq!(format!("{}", hexstr!(four_bytes)), "07A1FFC7");
150/// assert_eq!(format!("{:x}", hexstr!(four_bytes)), "07a1ffc7");
151/// ```
152macro_rules! hexstr {
153    ($array:expr) => {
154        $crate::hex_str!($array, sep: "")
155    }
156}

158#[allow(non_snake_case)]
159/// Explicitly construct a newtype to format with.
160///
161/// The first generic parameter specifies the block size in bytes, the second parameter
162/// the separator string.
163///
164/// This function is just here for documentation of what the `hex_str!` macro
165/// does internally (using the undocumented `typeint!` and `typesep!` macros).
166///
167/// ```
168/// use delog::hex::{HexStr, Separator, Unsigned};
169///
170/// // typeint!(U3, 3);
171/// struct U3 {}
172/// impl Unsigned for U3 {
173///     const N: usize = 3;
174/// }
175///
176/// // typesep!(Pipe, "|");
177/// struct Pipe {}
178/// impl Separator for Pipe {
179///     const SEPARATOR: &'static str  = "|";
180/// }
181///
182/// let four_bytes = &[7u8, 0xA1, 255, 0xC7];
183/// let hex_str = HexStr::<_, U3, Pipe>(four_bytes);
184///
185/// assert_eq!(format!("{}", hex_str), "07A1FF|C7");
186/// ```
187pub fn HexStr<T: ?Sized, U: Unsigned, S: Separator>(value: &T) -> HexStr<T, U, S> {
188    HexStr {
189        value,
190        _bytes_per_block: PhantomData,
191        _separator: PhantomData,
192    }
193}

219macro_rules! implement {
220    ($Trait:ident, $padded_formatter:expr) => {
221        impl<'a, T: ?Sized, U, S> fmt::$Trait for HexStr<'a, T, U, S>
222        where
223            T: AsRef<[u8]>,
224            U: Unsigned,
225            S: Separator,
226        {
227            fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
228                use core::fmt::{self, Alignment::*};
229                let max_bytes = f.width().unwrap_or(usize::MAX);
230                let bytes = self.value.as_ref();
231
232                #[inline]
233                fn nontruncated_fmt(
234                    bytes: &[u8],
235                    f: &mut fmt::Formatter<'_>,
236                    chunk_size: usize,
237                    separator: &str,
238                ) -> Result<(), fmt::Error> {
239                    let mut first = true;
240                    for entry in bytes.chunks(chunk_size) {
241                        if first {
242                            first = false;
243                        } else {
244                            f.write_str(separator)?;
245                        }
246                        for byte in entry.iter() {
247                            write!(f, $padded_formatter, byte)?;
248                        }
249                    }
250                    Ok(())
251                }
252
253                // const ELLIPSIS: &str = "…";
254                const ELLIPSIS: &str = "..";
255
256                let chunk_size = U::N;
257                let separator = S::SEPARATOR;
258
259                if bytes.len() <= max_bytes {
260                    nontruncated_fmt(bytes, f, chunk_size, separator)
261                } else {
262                    let align = f.align().unwrap_or(Center);
263                    let (left, right) = match align {
264                        Left => (max_bytes, 0),
265                        Center => (max_bytes - max_bytes / 2, max_bytes / 2),
266                        Right => (0, max_bytes),
267                    };
268                    nontruncated_fmt(&bytes[..left], f, chunk_size, separator)?;
269                    // if left > 0 {
270                    //     f.write_str(separator)?;
271                    // }
272                    f.write_str(ELLIPSIS)?;
273                    // if right > 0 {
274                    //     f.write_str(separator)?;
275                    // }
276                    nontruncated_fmt(&bytes[bytes.len() - right..], f, chunk_size, separator)?;
277                    Ok(())
278                }
279            }
280        }
281    };
282}
