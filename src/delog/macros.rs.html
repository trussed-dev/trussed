1/// Fallible (ungated) version of `log!`.
2#[macro_export]
3#[doc(hidden)]
4#[cfg(not(feature = "std-log"))]
5macro_rules! try_log {
6    (target: $target:expr, $lvl:expr, $($arg:tt)+) => ({
7        let lvl = $lvl;
8        if lvl <= $crate::log::STATIC_MAX_LEVEL && lvl <= $crate::log::max_level() {
9            $crate::__private_api_try_log(
10                ::core::format_args!($($arg)+),
11                lvl,
12                &($target, ::core::module_path!(), ::core::file!(), ::core::line!()),
13            )
14        } else {
15            Ok(())
16        }
17    });
18
19    ($lvl:expr, $($arg:tt)+) => ($crate::try_log!(target: ::core::module_path!(), $lvl, $($arg)+))
20}

30    });

38// There is a syntax issue with "repetitions in binding patterns for nested macros",
39// with a workaround: https://github.com/rust-lang/rust/issues/35853#issuecomment-443110660
40//
41// This is why we want to  have `($)` expressions in the following, can just imagine they're not there.
42//
43// Unfortunately, I couldn't get this to work, so instead we use the weird `with_dollar_sign!` instead.
44
45#[macro_export]
46#[doc(hidden)]
47macro_rules! with_dollar_sign {
48    ($($body:tt)*) => {
49        macro_rules! __with_dollar_sign { $($body)* }
50        __with_dollar_sign!($);
51    }
52}
