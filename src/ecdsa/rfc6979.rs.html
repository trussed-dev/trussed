1//! Support for computing deterministic ECDSA ephemeral scalar (`k`).
2//!
3//! Implementation of the algorithm described in RFC 6979 (Section 3.2):
4//! <https://tools.ietf.org/html/rfc6979#section-3>
5
6use crate::hazmat::FromDigest;
7use elliptic_curve::{
8    generic_array::GenericArray,
9    group::ff::PrimeField,
10    ops::Invert,
11    weierstrass::Curve,
12    zeroize::{Zeroize, Zeroizing},
13    FieldBytes, FieldSize, NonZeroScalar, ProjectiveArithmetic, Scalar,
14};
15use hmac::{Hmac, Mac, NewMac};
16use signature::digest::{BlockInput, FixedOutput, Reset, Update};
17
18/// Generate ephemeral scalar `k` from the secret scalar and a digest of the
19/// input message.
20pub fn generate_k<C, D>(
21    secret_scalar: &NonZeroScalar<C>,
22    msg_digest: D,
23    additional_data: &[u8],
24) -> Zeroizing<NonZeroScalar<C>>
25where
26    C: Curve + ProjectiveArithmetic,
27    D: FixedOutput<OutputSize = FieldSize<C>> + BlockInput + Clone + Default + Reset + Update,
28    Scalar<C>: FromDigest<C> + Invert<Output = Scalar<C>> + Zeroize,
29{
30    let mut x = secret_scalar.to_repr();
31    let h1 = Scalar::<C>::from_digest(msg_digest).to_repr();
32    let mut hmac_drbg = HmacDrbg::<D>::new(&x, &h1, additional_data);
33    x.zeroize();
34
35    loop {
36        let mut tmp = FieldBytes::<C>::default();
37        hmac_drbg.generate_into(&mut tmp);
38        if let Some(k) = NonZeroScalar::from_repr(tmp) {
39            return Zeroizing::new(k);
40        }
41    }
42}

44/// Internal implementation of `HMAC_DRBG` as described in NIST SP800-90A:
45/// <https://csrc.nist.gov/publications/detail/sp/800-90a/rev-1/final>
46///
47/// This is a HMAC-based deterministic random bit generator used internally
48/// to compute a deterministic ECDSA ephemeral scalar `k`.
49// TODO(tarcieri): use `hmac-drbg` crate when sorpaas/rust-hmac-drbg#3 is merged
50struct HmacDrbg<D>
51where
52    D: BlockInput + FixedOutput + Clone + Default + Reset + Update,
53{
54    /// HMAC key `K` (see RFC 6979 Section 3.2.c)
55    k: Hmac<D>,
56
57    /// Chaining value `V` (see RFC 6979 Section 3.2.c)
58    v: GenericArray<u8, D::OutputSize>,
59}
60
61impl<D> HmacDrbg<D>
62where
63    D: BlockInput + FixedOutput + Clone + Default + Reset + Update,
64{
65    /// Initialize `HMAC_DRBG`
66    pub fn new(entropy_input: &[u8], nonce: &[u8], additional_data: &[u8]) -> Self {
67        let mut k = Hmac::new(&Default::default());
68        let mut v = GenericArray::default();
69
70        for b in &mut v {
71            *b = 0x01;
72        }
73
74        for i in 0..=1 {
75            k.update(&v);
76            k.update(&[i]);
77            k.update(entropy_input);
78            k.update(nonce);
79            k.update(additional_data);
80            k = Hmac::new_from_slice(&k.finalize().into_bytes()).expect("HMAC error");
81
82            // Steps 3.2.e,g: v = HMAC_k(v)
83            k.update(&v);
84            v = k.finalize_reset().into_bytes();
85        }
86
87        Self { k, v }
88    }
89
90    /// Get the next `HMAC_DRBG` output
91    pub fn generate_into(&mut self, out: &mut [u8]) {
92        for out_chunk in out.chunks_mut(self.v.len()) {
93            self.k.update(&self.v);
94            self.v = self.k.finalize_reset().into_bytes();
95            out_chunk.copy_from_slice(&self.v[..out_chunk.len()]);
96        }
97
98        self.k.update(&self.v);
99        self.k.update(&[0x00]);
100        self.k = Hmac::new_from_slice(&self.k.finalize_reset().into_bytes()).expect("HMAC error");
101        self.k.update(&self.v);
102        self.v = self.k.finalize_reset().into_bytes();
103    }
104}

106#[cfg(test)]
107mod tests {
108    use super::generate_k;
109    use elliptic_curve::{dev::NonZeroScalar, group::ff::PrimeField};
110    use hex_literal::hex;
111    use sha2::{Digest, Sha256};
112
113    /// Test vector from RFC 6979 Appendix 2.5 (NIST P-256 + SHA-256)
114    /// <https://tools.ietf.org/html/rfc6979#appendix-A.2.5>
115    #[test]
116    fn appendix_2_5_test_vector() {
117        let x = NonZeroScalar::from_repr(
118            hex!("c9afa9d845ba75166b5c215767b1d6934e50c3db36e89b127b8a622b120f6721").into(),
119        )
120        .unwrap();
121
122        let digest = Sha256::new().chain("sample");
123        let k = generate_k(&x, digest, &[]);
124
125        assert_eq!(
126            k.to_repr().as_slice(),
127            &hex!("a6e3c57dd01abe90086538398355dd4c3b17aa873382b0f24d6129493d8aad60")[..]
128        );
129    }
