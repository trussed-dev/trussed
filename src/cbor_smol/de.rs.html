8/// Deserialize a message of type `T` from a byte slice. The unused portion (if any)
9/// of the byte slice is returned for further usage
10pub fn from_bytes<'a, T>(s: &'a [u8]) -> Result<T>
11where
12    T: Deserialize<'a>,
13{
14    let mut deserializer = Deserializer::from_bytes(s);
15    let t = T::deserialize(&mut deserializer)?;
16    Ok(t)
17}
18
19/// Deserialize a message of type `T` from a byte slice. The unused portion (if any)
20/// of the byte slice is returned for further usage
21pub fn take_from_bytes<'a, T>(s: &'a [u8]) -> Result<(T, &'a [u8])>
22where
23    T: Deserialize<'a>,
24{
25    let mut deserializer = Deserializer::from_bytes(s);
26    let t = T::deserialize(&mut deserializer)?;
27    Ok((t, deserializer.input))
28}

54impl<'de> Deserializer<'de> {
55    fn try_take_n(&mut self, count: usize) -> Result<&'de [u8]> {
56        if self.input.len() >= count {
57            let (a, b) = self.input.split_at(count);
58            self.input = b;
59            Ok(a)
60        } else {
61            Err(Error::DeserializeUnexpectedEnd)
62        }
63    }
64
65    fn peek_major(&mut self) -> Result<u8> {
66        if !self.input.is_empty() {
67            let byte = self.input[0];
68            Ok(byte >> MAJOR_OFFSET)
69        } else {
70            Err(Error::DeserializeUnexpectedEnd)
71        }
72    }
73
74    fn peek(&mut self) -> Result<u8> {
75        if !self.input.is_empty() {
76            Ok(self.input[0])
77        } else {
78            Err(Error::DeserializeUnexpectedEnd)
79        }
80    }
81
82    fn consume(&mut self) -> Result<u8> {
83        if !self.input.is_empty() {
84            let ret = self.input[0];
85            self.input = &self.input[1..];
86            Ok(ret)
87        } else {
88            Err(Error::DeserializeUnexpectedEnd)
89        }
90    }
91
92    fn expect_major(&mut self, major: u8) -> Result<u8> {
93        let byte = self.try_take_n(1)?[0];
94        if major != (byte >> MAJOR_OFFSET) {
95            // logging::info_now!("expecting {}, got {} in byte {}", major, byte >> 5, byte).ok();
96            // logging::info_now!("remaining data: {:?}", &self.input).ok();
97            return Err(Error::DeserializeBadMajor);
98        }
99        Ok(byte & ((1 << MAJOR_OFFSET) - 1))
100    }
101
102    // TODO: name something like "one-byte-integer"
103    fn raw_deserialize_u8(&mut self, major: u8) -> Result<u8> {
104        let additional = self.expect_major(major)?;
105
106        match additional {
107            byte @ 0..=23 => Ok(byte),
108            24 => match self.try_take_n(1)?[0] {
109                0..=23 => Err(Error::DeserializeNonMinimal),
110                byte => Ok(byte),
111            },
112            _ => Err(Error::DeserializeBadU8),
113        }
114    }
115
116    fn raw_deserialize_u16(&mut self, major: u8) -> Result<u16> {
117        let number = self.raw_deserialize_u32(major)?;
118        if number <= u16::MAX as u32 {
119            Ok(number as u16)
120        } else {
121            Err(Error::DeserializeBadU16)
122        }
123    }
124
125    fn raw_deserialize_u32(&mut self, major: u8) -> Result<u32> {
126        let additional = self.expect_major(major)?;
127
128        match additional {
129            byte @ 0..=23 => Ok(byte as u32),
130            24 => match self.try_take_n(1)?[0] {
131                0..=23 => Err(Error::DeserializeNonMinimal),
132                byte => Ok(byte as u32),
133            },
134            25 => {
135                let unsigned = u16::from_be_bytes(
136                    self.try_take_n(2)?
137                        .try_into()
138                        .map_err(|_| Error::InexistentSliceToArrayError)?,
139                );
140                match unsigned {
141                    0..=255 => Err(Error::DeserializeNonMinimal),
142                    unsigned => Ok(unsigned as u32),
143                }
144            }
145            26 => {
146                let unsigned = u32::from_be_bytes(
147                    self.try_take_n(4)?
148                        .try_into()
149                        .map_err(|_| Error::InexistentSliceToArrayError)?,
150                );
151                match unsigned {
152                    0..=65535 => Err(Error::DeserializeNonMinimal),
153                    unsigned => Ok(unsigned),
154                }
155            }
156            _ => Err(Error::DeserializeBadU32),
157        }
158    }
159    fn raw_deserialize_u64(&mut self, major: u8) -> Result<u64> {
160        let additional = self.expect_major(major)?;
161
162        match additional {
163            byte @ 0..=23 => Ok(byte as u64),
164            24 => match self.try_take_n(1)?[0] {
165                0..=23 => Err(Error::DeserializeNonMinimal),

174                match unsigned {

177                }
178            }
179            26 => {
180                let unsigned = u32::from_be_bytes(
181                    self.try_take_n(4)?
182                        .try_into()
183                        .map_err(|_| Error::InexistentSliceToArrayError)?,
184                );
185                match unsigned {
186                    0..=65535 => Err(Error::DeserializeNonMinimal),
187                    unsigned => Ok(unsigned as u64),
188                }
189            }
190            27 => {
191                let unsigned = u64::from_be_bytes(
192                    self.try_take_n(8)?
193                        .try_into()
194                        .map_err(|_| Error::InexistentSliceToArrayError)?,
195                );
196                match unsigned {
197                    0..=0xFFFFFFFF => Err(Error::DeserializeNonMinimal),
198                    unsigned => Ok(unsigned),
199                }
200            }
201            _ => Err(Error::DeserializeBadU64),
202        }
203    }
204
205    fn ignore_int(&mut self, major: u8) -> Result<()> {
206        let additional = self.expect_major(major)?;
207        match additional {
208            0..=23 => {}
209            24 => {
210                self.try_take_n(1)?;
211            }
212            25 => {
213                self.try_take_n(2)?;
214            }
215            26 => {
216                self.try_take_n(4)?;
217            }
218            27 => {
219                self.try_take_n(8)?;
220            }
221            _ => return Err(Error::DeserializeBadU16),
222        };
223        Ok(())
224    }
225
226    fn ignore_bytes(&mut self, major: u8) -> Result<()> {
227        let length = self.raw_deserialize_u32(major)? as usize;
228        self.try_take_n(length)?;
229        Ok(())
230    }
231
232    fn ignore_array(&mut self, major: u8, mult: usize) -> Result<()> {
233        let length = self.raw_deserialize_u32(major)? as usize;
234        let Some(real_length) = length.checked_mul(mult) else {
235            return Err(Error::InexistentSliceToArrayError);
236        };
237        for _ in 0..real_length {
238            self.ignore()?;
239        }
240        Ok(())
241    }
242
243    fn ignore_float(&mut self) -> Result<()> {
244        let additional = self.expect_major(7)?;
245        match additional {
246            0..=23 => {}
247            24 => {
248                self.try_take_n(1)?;
249            }
250            25 => {
251                self.try_take_n(2)?;
252            }
253            26 => {
254                self.try_take_n(4)?;
255            }
256            27 => {
257                self.try_take_n(8)?;
258            }
259            _ => return Err(Error::DeserializeBadMajor),
260        };
261        Ok(())
262    }
263
264    fn ignore(&mut self) -> Result<()> {
265        let major = self.peek_major()?;
266        match major {
267            MAJOR_POSINT | MAJOR_NEGINT => self.ignore_int(major)?,
268            MAJOR_BYTES | MAJOR_STR => self.ignore_bytes(major)?,
269            MAJOR_ARRAY => self.ignore_array(MAJOR_ARRAY, 1)?,
270            MAJOR_MAP => self.ignore_array(MAJOR_MAP, 2)?,
271            6 => {
272                self.ignore_int(6)?;
273                self.ignore()?;
274            }
275            MAJOR_FLOAT => self.ignore_float()?,
276            _ => return Err(Error::DeserializeBadMajor),
277        }
278        Ok(())
279    }

362impl<'de, 'a> serde::de::VariantAccess<'de> for EnumAccess<'a, 'de> {
363    type Error = Error;
364
365    fn unit_variant(self) -> Result<()> {
366        if self.variant_len != 0 {
367            return Err(Error::DeserializeBadEnum);
368        }
369        Ok(())
370    }
371
372    fn newtype_variant_seed<V: DeserializeSeed<'de>>(self, seed: V) -> Result<V::Value> {
373        if 2 != self.variant_len {
374            return Err(Error::DeserializeBadEnum);
375        }
376        DeserializeSeed::deserialize(seed, self.deserializer)
377    }
378
379    fn tuple_variant<V: Visitor<'de>>(self, len: usize, visitor: V) -> Result<V::Value> {
380        if len + 1 != self.variant_len {
381            return Err(Error::DeserializeBadEnum);
382        }
383
384        visitor.visit_seq(SeqAccess {
385            deserializer: self.deserializer,
386            len,
387        })
388    }
389
390    fn struct_variant<V: Visitor<'de>>(
391        self,
392        _fields: &'static [&'static str],
393        visitor: V,
394    ) -> Result<V::Value> {
395        if 2 != self.variant_len {
396            return Err(Error::DeserializeBadEnum);
397        }
398        serde::de::Deserializer::deserialize_map(self.deserializer, visitor)
399    }
400}

776    fn deserialize_enum<V>(
777        self,
778        _name: &'static str,
779        _variants: &'static [&'static str],
780        visitor: V,
781    ) -> Result<V::Value>
782    where
783        V: Visitor<'de>,
784    {
785        match self.peek_major()? {
786            // Data variant
787            MAJOR_ARRAY => {
788                let len = self.raw_deserialize_u32(MAJOR_ARRAY)?;
789                visitor.visit_enum(EnumAccess {
790                    deserializer: self,
791                    variant_len: len as usize,
792                })
793            }
794            // Unit variant
795            MAJOR_POSINT => visitor.visit_enum(EnumAccess {
796                deserializer: self,
797                variant_len: 0,
798            }),
799            _ => Err(Error::DeserializeBadMajor),
800        }
801    }
802
803    fn deserialize_identifier<V>(self, visitor: V) -> Result<V::Value>
804    where
805        V: Visitor<'de>,
806    {
807        let major = self.peek_major()?;
808        match major {
809            MAJOR_BYTES | MAJOR_STR => {
810                // Rust identifiers are always valid UTF-8 so we can assume that bytes are
811                // UTF-8-encoded strings.  This has the benefit that we only need a mapping from
812                // strings to fields (and the mapping from bytes to fields can be optimized out).
813                let length = self.raw_deserialize_u32(major)? as usize;
814                let bytes: &'de [u8] = self.try_take_n(length)?;
815                let string_slice =
816                    core::str::from_utf8(bytes).map_err(|_| Error::DeserializeBadUtf8)?;
817                visitor.visit_borrowed_str(string_slice)
818            }
819            MAJOR_POSINT => self.deserialize_u64(visitor),
820            _ => Err(Error::DeserializeBadMajor),
821        }
822    }

824    fn deserialize_ignored_any<V>(self, visitor: V) -> Result<V::Value>
825    where
826        V: Visitor<'de>,
827    {
828        // Ignore extra fields/options
829        self.ignore()?;
830        visitor.visit_none()
831    }
