17/// The number of 64-bit limbs used to represent a [`FieldElement`].
18const LIMBS: usize = 4;
19
20/// Constant representing the modulus
21/// p = 2^{224}(2^{32} − 1) + 2^{192} + 2^{96} − 1
22pub const MODULUS: FieldElement = FieldElement([
23    0xffff_ffff_ffff_ffff,
24    0x0000_0000_ffff_ffff,
25    0x0000_0000_0000_0000,
26    0xffff_ffff_0000_0001,
27]);
28
29/// R = 2^256 mod p
30const R: FieldElement = FieldElement([
31    0x0000_0000_0000_0001,
32    0xffff_ffff_0000_0000,
33    0xffff_ffff_ffff_ffff,
34    0x0000_0000_ffff_fffe,
35]);
36
37/// R^2 = 2^512 mod p
38const R2: FieldElement = FieldElement([
39    0x0000_0000_0000_0003,
40    0xffff_fffb_ffff_ffff,
41    0xffff_ffff_ffff_fffe,
42    0x0000_0004_ffff_fffd,
43]);

45/// An element in the finite field modulo p = 2^{224}(2^{32} − 1) + 2^{192} + 2^{96} − 1.
46// The internal representation is in little-endian order. Elements are always in
47// Montgomery form; i.e., FieldElement(a) = aR mod p, with R = 2^256.
48#[derive(Clone, Copy, Debug)]
49pub struct FieldElement(pub(crate) [u64; LIMBS]);
50
51impl ConditionallySelectable for FieldElement {
52    fn conditional_select(a: &FieldElement, b: &FieldElement, choice: Choice) -> FieldElement {
53        FieldElement([
54            u64::conditional_select(&a.0[0], &b.0[0], choice),
55            u64::conditional_select(&a.0[1], &b.0[1], choice),
56            u64::conditional_select(&a.0[2], &b.0[2], choice),
57            u64::conditional_select(&a.0[3], &b.0[3], choice),
58        ])
59    }
60}

66            & self.0[2].ct_eq(&other.0[2])
67            & self.0[3].ct_eq(&other.0[3])

94    /// Returns a uniformly-random element within the field.
95    pub fn generate(mut rng: impl CryptoRng + RngCore) -> Self {
96        // We reduce a random 512-bit value into a 256-bit field, which results in a
97        // negligible bias from the uniform distribution.
98        let mut buf = [0; 64];
99        rng.fill_bytes(&mut buf);
100        FieldElement::from_bytes_wide(buf)
101    }

103    fn from_bytes_wide(bytes: [u8; 64]) -> Self {
104        FieldElement::montgomery_reduce(
105            u64::from_be_bytes(bytes[0..8].try_into().unwrap()),
106            u64::from_be_bytes(bytes[8..16].try_into().unwrap()),
107            u64::from_be_bytes(bytes[16..24].try_into().unwrap()),
108            u64::from_be_bytes(bytes[24..32].try_into().unwrap()),
109            u64::from_be_bytes(bytes[32..40].try_into().unwrap()),
110            u64::from_be_bytes(bytes[40..48].try_into().unwrap()),
111            u64::from_be_bytes(bytes[48..56].try_into().unwrap()),
112            u64::from_be_bytes(bytes[56..64].try_into().unwrap()),
113        )
114    }

116    /// Attempts to parse the given byte array as an SEC1-encoded field element.
117    ///
118    /// Returns None if the byte array does not contain a big-endian integer in the range
119    /// [0, p).
120    pub fn from_bytes(bytes: &FieldBytes) -> CtOption<Self> {
121        let mut w = [0u64; LIMBS];
122
123        // Interpret the bytes as a big-endian integer w.
124        w[3] = u64::from_be_bytes(bytes[0..8].try_into().unwrap());
125        w[2] = u64::from_be_bytes(bytes[8..16].try_into().unwrap());
126        w[1] = u64::from_be_bytes(bytes[16..24].try_into().unwrap());
127        w[0] = u64::from_be_bytes(bytes[24..32].try_into().unwrap());
128
129        // If w is in the range [0, p) then w - p will overflow, resulting in a borrow
130        // value of 2^64 - 1.
131        let (_, borrow) = sbb(w[0], MODULUS.0[0], 0);
132        let (_, borrow) = sbb(w[1], MODULUS.0[1], borrow);
133        let (_, borrow) = sbb(w[2], MODULUS.0[2], borrow);
134        let (_, borrow) = sbb(w[3], MODULUS.0[3], borrow);
135        let is_some = (borrow as u8) & 1;
136
137        // Convert w to Montgomery form: w * R^2 * R^-1 mod p = wR mod p
138        CtOption::new(FieldElement(w).mul(&R2), Choice::from(is_some))
139    }

141    /// Returns the SEC1 encoding of this field element.
142    pub fn to_bytes(&self) -> FieldBytes {
143        // Convert from Montgomery form to canonical form
144        let tmp =
145            FieldElement::montgomery_reduce(self.0[0], self.0[1], self.0[2], self.0[3], 0, 0, 0, 0);
146
147        let mut ret = FieldBytes::default();
148        ret[0..8].copy_from_slice(&tmp.0[3].to_be_bytes());
149        ret[8..16].copy_from_slice(&tmp.0[2].to_be_bytes());
150        ret[16..24].copy_from_slice(&tmp.0[1].to_be_bytes());
151        ret[24..32].copy_from_slice(&tmp.0[0].to_be_bytes());
152        ret
153    }

174    /// Returns self + rhs mod p
175    pub const fn add(&self, rhs: &Self) -> Self {
176        // Bit 256 of p is set, so addition can result in five words.
177        let (w0, carry) = adc(self.0[0], rhs.0[0], 0);
178        let (w1, carry) = adc(self.0[1], rhs.0[1], carry);
179        let (w2, carry) = adc(self.0[2], rhs.0[2], carry);
180        let (w3, w4) = adc(self.0[3], rhs.0[3], carry);
181
182        // Attempt to subtract the modulus, to ensure the result is in the field.
183        Self::sub_inner(
184            w0,
185            w1,
186            w2,
187            w3,
188            w4,
189            MODULUS.0[0],
190            MODULUS.0[1],
191            MODULUS.0[2],
192            MODULUS.0[3],
193            0,
194        )
195    }

197    /// Returns 2*self.
198    pub const fn double(&self) -> Self {
199        self.add(self)
200    }

202    /// Returns self - rhs mod p
203    pub const fn subtract(&self, rhs: &Self) -> Self {
204        Self::sub_inner(
205            self.0[0], self.0[1], self.0[2], self.0[3], 0, rhs.0[0], rhs.0[1], rhs.0[2], rhs.0[3],
206            0,
207        )
208    }

210    #[inline]
211    #[allow(clippy::too_many_arguments)]
212    const fn sub_inner(
213        l0: u64,
214        l1: u64,
215        l2: u64,
216        l3: u64,
217        l4: u64,
218        r0: u64,
219        r1: u64,
220        r2: u64,
221        r3: u64,
222        r4: u64,
223    ) -> Self {
224        let (w0, borrow) = sbb(l0, r0, 0);
225        let (w1, borrow) = sbb(l1, r1, borrow);
226        let (w2, borrow) = sbb(l2, r2, borrow);
227        let (w3, borrow) = sbb(l3, r3, borrow);
228        let (_, borrow) = sbb(l4, r4, borrow);
229
230        // If underflow occurred on the final limb, borrow = 0xfff...fff, otherwise
231        // borrow = 0x000...000. Thus, we use it as a mask to conditionally add the
232        // modulus.
233        let (w0, carry) = adc(w0, MODULUS.0[0] & borrow, 0);
234        let (w1, carry) = adc(w1, MODULUS.0[1] & borrow, carry);
235        let (w2, carry) = adc(w2, MODULUS.0[2] & borrow, carry);
236        let (w3, _) = adc(w3, MODULUS.0[3] & borrow, carry);
237
238        FieldElement([w0, w1, w2, w3])
239    }

241    /// Montgomery Reduction
242    ///
243    /// The general algorithm is:
244    /// ```text
245    /// A <- input (2n b-limbs)
246    /// for i in 0..n {
247    ///     k <- A[i] p' mod b
248    ///     A <- A + k p b^i
249    /// }
250    /// A <- A / b^n
251    /// if A >= p {
252    ///     A <- A - p
253    /// }
254    /// ```
255    ///
256    /// For secp256r1, we have the following simplifications:
257    ///
258    /// - `p'` is 1, so our multiplicand is simply the first limb of the intermediate A.
259    ///
260    /// - The first limb of p is 2^64 - 1; multiplications by this limb can be simplified
261    ///   to a shift and subtraction:
262    ///   ```text
263    ///       a_i * (2^64 - 1) = a_i * 2^64 - a_i = (a_i << 64) - a_i
264    ///   ```
265    ///   However, because `p' = 1`, the first limb of p is multiplied by limb i of the
266    ///   intermediate A and then immediately added to that same limb, so we simply
267    ///   initialize the carry to limb i of the intermediate.
268    ///
269    /// - The third limb of p is zero, so we can ignore any multiplications by it and just
270    ///   add the carry.
271    ///
272    /// References:
273    /// - Handbook of Applied Cryptography, Chapter 14
274    ///   Algorithm 14.32
275    ///   http://cacr.uwaterloo.ca/hac/about/chap14.pdf
276    ///
277    /// - Efficient and Secure Elliptic Curve Cryptography Implementation of Curve P-256
278    ///   Algorithm 7) Montgomery Word-by-Word Reduction
279    ///   https://csrc.nist.gov/csrc/media/events/workshop-on-elliptic-curve-cryptography-standards/documents/papers/session6-adalier-mehmet.pdf
280    #[inline]
281    #[allow(clippy::too_many_arguments)]
282    const fn montgomery_reduce(
283        r0: u64,
284        r1: u64,
285        r2: u64,
286        r3: u64,
287        r4: u64,
288        r5: u64,
289        r6: u64,
290        r7: u64,
291    ) -> Self {
292        let (r1, carry) = mac(r1, r0, MODULUS.0[1], r0);
293        let (r2, carry) = adc(r2, 0, carry);
294        let (r3, carry) = mac(r3, r0, MODULUS.0[3], carry);
295        let (r4, carry2) = adc(r4, 0, carry);
296
297        let (r2, carry) = mac(r2, r1, MODULUS.0[1], r1);
298        let (r3, carry) = adc(r3, 0, carry);
299        let (r4, carry) = mac(r4, r1, MODULUS.0[3], carry);
300        let (r5, carry2) = adc(r5, carry2, carry);
301
302        let (r3, carry) = mac(r3, r2, MODULUS.0[1], r2);
303        let (r4, carry) = adc(r4, 0, carry);
304        let (r5, carry) = mac(r5, r2, MODULUS.0[3], carry);
305        let (r6, carry2) = adc(r6, carry2, carry);
306
307        let (r4, carry) = mac(r4, r3, MODULUS.0[1], r3);
308        let (r5, carry) = adc(r5, 0, carry);
309        let (r6, carry) = mac(r6, r3, MODULUS.0[3], carry);
310        let (r7, r8) = adc(r7, carry2, carry);
311
312        // Result may be within MODULUS of the correct value
313        Self::sub_inner(
314            r4,
315            r5,
316            r6,
317            r7,
318            r8,
319            MODULUS.0[0],
320            MODULUS.0[1],
321            MODULUS.0[2],
322            MODULUS.0[3],
323            0,
324        )
325    }

327    /// Returns self * rhs mod p
328    pub const fn mul(&self, rhs: &Self) -> Self {
329        // Schoolbook multiplication.
330
331        let (w0, carry) = mac(0, self.0[0], rhs.0[0], 0);
332        let (w1, carry) = mac(0, self.0[0], rhs.0[1], carry);
333        let (w2, carry) = mac(0, self.0[0], rhs.0[2], carry);
334        let (w3, w4) = mac(0, self.0[0], rhs.0[3], carry);
335
336        let (w1, carry) = mac(w1, self.0[1], rhs.0[0], 0);
337        let (w2, carry) = mac(w2, self.0[1], rhs.0[1], carry);
338        let (w3, carry) = mac(w3, self.0[1], rhs.0[2], carry);
339        let (w4, w5) = mac(w4, self.0[1], rhs.0[3], carry);
340
341        let (w2, carry) = mac(w2, self.0[2], rhs.0[0], 0);
342        let (w3, carry) = mac(w3, self.0[2], rhs.0[1], carry);
343        let (w4, carry) = mac(w4, self.0[2], rhs.0[2], carry);
344        let (w5, w6) = mac(w5, self.0[2], rhs.0[3], carry);
345
346        let (w3, carry) = mac(w3, self.0[3], rhs.0[0], 0);
347        let (w4, carry) = mac(w4, self.0[3], rhs.0[1], carry);
348        let (w5, carry) = mac(w5, self.0[3], rhs.0[2], carry);
349        let (w6, w7) = mac(w6, self.0[3], rhs.0[3], carry);
350
351        FieldElement::montgomery_reduce(w0, w1, w2, w3, w4, w5, w6, w7)
352    }

360    /// Returns `self^by`, where `by` is a little-endian integer exponent.
361    ///
362    /// **This operation is variable time with respect to the exponent.** If the exponent
363    /// is fixed, this operation is effectively constant time.
364    pub fn pow_vartime(&self, by: &[u64; 4]) -> Self {
365        let mut res = Self::one();
366        for e in by.iter().rev() {
367            for i in (0..64).rev() {
368                res = res.square();
369
370                if ((*e >> i) & 1) == 1 {
371                    res = res * self;
372                }
373            }
374        }
375        res
376    }

378    /// Returns the multiplicative inverse of self, if self is non-zero.
379    pub fn invert(&self) -> CtOption<Self> {
380        // We need to find b such that b * a ≡ 1 mod p. As we are in a prime
381        // field, we can apply Fermat's Little Theorem:
382        //
383        //    a^p         ≡ a mod p
384        //    a^(p-1)     ≡ 1 mod p
385        //    a^(p-2) * a ≡ 1 mod p
386        //
387        // Thus inversion can be implemented with a single exponentiation.
388        let inverse = self.pow_vartime(&[
389            0xffff_ffff_ffff_fffd,
390            0x0000_0000_ffff_ffff,
391            0x0000_0000_0000_0000,
392            0xffff_ffff_0000_0001,
393        ]);
394
395        CtOption::new(inverse, !self.is_zero())
396    }

398    /// Returns the square root of self mod p, or `None` if no square root exists.
399    pub fn sqrt(&self) -> CtOption<Self> {
400        // We need to find alpha such that alpha^2 = beta mod p. For secp256r1,
401        // p ≡ 3 mod 4. By Euler's Criterion, beta^(p-1)/2 ≡ 1 mod p. So:
402        //
403        //     alpha^2 = beta beta^((p - 1) / 2) mod p ≡ beta^((p + 1) / 2) mod p
404        //     alpha = ± beta^((p + 1) / 4) mod p
405        //
406        // Thus sqrt can be implemented with a single exponentiation.
407
408        let sqrt = self.pow_vartime(&[
409            0x0000_0000_0000_0000,
410            0x0000_0000_4000_0000,
411            0x4000_0000_0000_0000,
412            0x3fff_ffff_c000_0000,
413        ]);
414
415        CtOption::new(
416            sqrt,
417            (&sqrt * &sqrt).ct_eq(self), // Only return Some if it's the square root.
418        )
419    }
