80/// Scalars are elements in the finite field modulo n.
81///
82/// # Trait impls
83///
84/// Much of the important functionality of scalars is provided by traits from
85/// the [`ff`](https://docs.rs/ff/) crate, which is re-exported as
86/// `p256::elliptic_curve::ff`:
87///
88/// - [`Field`](https://docs.rs/ff/latest/ff/trait.Field.html) -
89///   represents elements of finite fields and provides:
90///   - [`Field::random`](https://docs.rs/ff/latest/ff/trait.Field.html#tymethod.random) -
91///     generate a random scalar
92///   - `double`, `square`, and `invert` operations
93///   - Bounds for [`Add`], [`Sub`], [`Mul`], and [`Neg`] (as well as `*Assign` equivalents)
94///   - Bounds for [`ConditionallySelectable`] from the `subtle` crate
95/// - [`PrimeField`](https://docs.rs/ff/0.9.0/ff/trait.PrimeField.html) -
96///   represents elements of prime fields and provides:
97///   - `from_repr`/`to_repr` for converting field elements from/to big integers.
98///   - `char_le_bits`, `multiplicative_generator`, `root_of_unity` constants.
99/// - [`PrimeFieldBits`](https://docs.rs/ff/latest/ff/trait.PrimeFieldBits.html) -
100///   operations over field elements represented as bits (requires `bits` feature)
101///
102/// Please see the documentation for the relevant traits for more information.
103#[derive(Clone, Copy, Debug, Default)]
104#[cfg_attr(docsrs, doc(cfg(feature = "arithmetic")))]
105pub struct Scalar(pub(crate) U256);

107impl Field for Scalar {
108    fn random(mut rng: impl RngCore) -> Self {
109        let mut bytes = FieldBytes::default();
110
111        // Generate a uniformly random scalar using rejection sampling,
112        // which produces a uniformly random distribution of scalars.
113        //
114        // This method is not constant time, but should be secure so long as
115        // rejected RNG outputs are unrelated to future ones (which is a
116        // necessary property of a `CryptoRng`).
117        //
118        // With an unbiased RNG, the probability of failing to complete after 4
119        // iterations is vanishingly small.
120        loop {
121            rng.fill_bytes(&mut bytes);
122            if let Some(scalar) = Scalar::from_repr(bytes) {
123                return scalar;
124            }
125        }
126    }

130    }

167    /// Attempts to parse the given byte array as an SEC1-encoded scalar.
168    ///
169    /// Returns None if the byte array does not contain a big-endian integer in the range
170    /// [0, p).
171    fn from_repr(bytes: FieldBytes) -> Option<Self> {
172        let mut w = [0u64; LIMBS];
173
174        // Interpret the bytes as a big-endian integer w.
175        w[3] = u64::from_be_bytes(bytes[0..8].try_into().unwrap());
176        w[2] = u64::from_be_bytes(bytes[8..16].try_into().unwrap());
177        w[1] = u64::from_be_bytes(bytes[16..24].try_into().unwrap());
178        w[0] = u64::from_be_bytes(bytes[24..32].try_into().unwrap());
179
180        // If w is in the range [0, n) then w - n will overflow, resulting in a borrow
181        // value of 2^64 - 1.
182        let (_, borrow) = sbb(w[0], MODULUS[0], 0);
183        let (_, borrow) = sbb(w[1], MODULUS[1], borrow);
184        let (_, borrow) = sbb(w[2], MODULUS[2], borrow);
185        let (_, borrow) = sbb(w[3], MODULUS[3], borrow);
186        let is_some = (borrow as u8) & 1;
187
188        CtOption::new(Scalar(w), Choice::from(is_some)).into()
189    }
190
191    fn to_repr(&self) -> FieldBytes {
192        self.to_bytes()
193    }
194
195    fn is_odd(&self) -> bool {
196        self.0[0] as u8 == 1
197    }

201    }

267/// Returns sign of left - right
268fn cmp_vartime(left: &U256, right: &U256) -> i32 {
269    use core::cmp::Ordering::*;
270
271    // since we're little-endian, need to reverse iterate
272    for (l, r) in left.iter().rev().zip(right.iter().rev()) {
273        match l.cmp(r) {
274            Less => return -1,
275            Greater => return 1,
276            Equal => continue,
277        }
278    }
279    0
280}
281
282fn shr1(u256: &mut U256) {
283    let mut bit: u64 = 0;
284    for digit in u256.iter_mut().rev() {
285        let new_digit = (bit << 63) | (*digit >> 1);
286        bit = *digit & 1;
287        *digit = new_digit;
288    }
289}

299        }

364    /// Returns self + rhs mod n
365    pub const fn add(&self, rhs: &Self) -> Self {
366        // Bit 256 of n is set, so addition can result in five words.
367        let (w0, carry) = adc(self.0[0], rhs.0[0], 0);
368        let (w1, carry) = adc(self.0[1], rhs.0[1], carry);
369        let (w2, carry) = adc(self.0[2], rhs.0[2], carry);
370        let (w3, w4) = adc(self.0[3], rhs.0[3], carry);
371
372        // Attempt to subtract the modulus, to ensure the result is in the field.
373        Self::sub_inner(
374            w0, w1, w2, w3, w4, MODULUS[0], MODULUS[1], MODULUS[2], MODULUS[3], 0,
375        )
376    }

391    #[inline]
392    #[allow(clippy::too_many_arguments)]
393    const fn sub_inner(
394        l0: u64,
395        l1: u64,
396        l2: u64,
397        l3: u64,
398        l4: u64,
399        r0: u64,
400        r1: u64,
401        r2: u64,
402        r3: u64,
403        r4: u64,
404    ) -> Self {
405        let (w0, borrow) = sbb(l0, r0, 0);
406        let (w1, borrow) = sbb(l1, r1, borrow);
407        let (w2, borrow) = sbb(l2, r2, borrow);
408        let (w3, borrow) = sbb(l3, r3, borrow);
409        let (_, borrow) = sbb(l4, r4, borrow);
410
411        // If underflow occurred on the final limb, borrow = 0xfff...fff, otherwise
412        // borrow = 0x000...000. Thus, we use it as a mask to conditionally add the
413        // modulus.
414        let (w0, carry) = adc(w0, MODULUS[0] & borrow, 0);
415        let (w1, carry) = adc(w1, MODULUS[1] & borrow, carry);
416        let (w2, carry) = adc(w2, MODULUS[2] & borrow, carry);
417        let (w3, _) = adc(w3, MODULUS[3] & borrow, carry);
418
419        Scalar([w0, w1, w2, w3])
420    }

422    /// Barrett Reduction
423    ///
424    /// The general algorithm is:
425    /// ```text
426    /// p = n = order of group
427    /// b = 2^64 = 64bit machine word
428    /// k = 4
429    /// a \in [0, 2^512]
430    /// mu := floor(b^{2k} / p)
431    /// q1 := floor(a / b^{k - 1})
432    /// q2 := q1 * mu
433    /// q3 := <- floor(a / b^{k - 1})
434    /// r1 := a mod b^{k + 1}
435    /// r2 := q3 * m mod b^{k + 1}
436    /// r := r1 - r2
437    ///
438    /// if r < 0: r := r + b^{k + 1}
439    /// while r >= p: do r := r - p (at most twice)
440    /// ```
441    ///
442    /// References:
443    /// - Handbook of Applied Cryptography, Chapter 14
444    ///   Algorithm 14.42
445    ///   http://cacr.uwaterloo.ca/hac/about/chap14.pdf
446    ///
447    /// - Efficient and Secure Elliptic Curve Cryptography Implementation of Curve P-256
448    ///   Algorithm 6) Barrett Reduction modulo p
449    ///   https://csrc.nist.gov/csrc/media/events/workshop-on-elliptic-curve-cryptography-standards/documents/papers/session6-adalier-mehmet.pdf
450    #[inline]
451    #[allow(clippy::too_many_arguments)]
452    const fn barrett_reduce(
453        a0: u64,
454        a1: u64,
455        a2: u64,
456        a3: u64,
457        a4: u64,
458        a5: u64,
459        a6: u64,
460        a7: u64,
461    ) -> Self {
462        let q1: [u64; 5] = [a3, a4, a5, a6, a7];
463
464        const fn q1_times_mu_shift_five(q1: &[u64; 5]) -> [u64; 5] {
465            // Schoolbook multiplication.
466
467            let (_w0, carry) = mac(0, q1[0], MU[0], 0);
468            let (w1, carry) = mac(0, q1[0], MU[1], carry);
469            let (w2, carry) = mac(0, q1[0], MU[2], carry);
470            let (w3, carry) = mac(0, q1[0], MU[3], carry);
471            let (w4, w5) = mac(0, q1[0], MU[4], carry);
472
473            let (_w1, carry) = mac(w1, q1[1], MU[0], 0);
474            let (w2, carry) = mac(w2, q1[1], MU[1], carry);
475            let (w3, carry) = mac(w3, q1[1], MU[2], carry);
476            let (w4, carry) = mac(w4, q1[1], MU[3], carry);
477            let (w5, w6) = mac(w5, q1[1], MU[4], carry);
478
479            let (_w2, carry) = mac(w2, q1[2], MU[0], 0);
480            let (w3, carry) = mac(w3, q1[2], MU[1], carry);
481            let (w4, carry) = mac(w4, q1[2], MU[2], carry);
482            let (w5, carry) = mac(w5, q1[2], MU[3], carry);
483            let (w6, w7) = mac(w6, q1[2], MU[4], carry);
484
485            let (_w3, carry) = mac(w3, q1[3], MU[0], 0);
486            let (w4, carry) = mac(w4, q1[3], MU[1], carry);
487            let (w5, carry) = mac(w5, q1[3], MU[2], carry);
488            let (w6, carry) = mac(w6, q1[3], MU[3], carry);
489            let (w7, w8) = mac(w7, q1[3], MU[4], carry);
490
491            let (_w4, carry) = mac(w4, q1[4], MU[0], 0);
492            let (w5, carry) = mac(w5, q1[4], MU[1], carry);
493            let (w6, carry) = mac(w6, q1[4], MU[2], carry);
494            let (w7, carry) = mac(w7, q1[4], MU[3], carry);
495            let (w8, w9) = mac(w8, q1[4], MU[4], carry);
496
497            // let q2 = [_w0, _w1, _w2, _w3, _w4, w5, w6, w7, w8, w9];
498            [w5, w6, w7, w8, w9]
499        }
500
501        let q3 = q1_times_mu_shift_five(&q1);
502
503        let r1: [u64; 5] = [a0, a1, a2, a3, a4];
504
505        const fn q3_times_n_keep_five(q3: &[u64; 5]) -> [u64; 5] {
506            // Schoolbook multiplication.
507
508            let (w0, carry) = mac(0, q3[0], MODULUS[0], 0);
509            let (w1, carry) = mac(0, q3[0], MODULUS[1], carry);
510            let (w2, carry) = mac(0, q3[0], MODULUS[2], carry);
511            let (w3, carry) = mac(0, q3[0], MODULUS[3], carry);
512            let (w4, _) = mac(0, q3[0], 0, carry);
513
514            let (w1, carry) = mac(w1, q3[1], MODULUS[0], 0);
515            let (w2, carry) = mac(w2, q3[1], MODULUS[1], carry);
516            let (w3, carry) = mac(w3, q3[1], MODULUS[2], carry);
517            let (w4, _) = mac(w4, q3[1], MODULUS[3], carry);
518
519            let (w2, carry) = mac(w2, q3[2], MODULUS[0], 0);
520            let (w3, carry) = mac(w3, q3[2], MODULUS[1], carry);
521            let (w4, _) = mac(w4, q3[2], MODULUS[2], carry);
522
523            let (w3, carry) = mac(w3, q3[3], MODULUS[0], 0);
524            let (w4, _) = mac(w4, q3[3], MODULUS[1], carry);
525
526            let (w4, _) = mac(w4, q3[4], MODULUS[0], 0);
527
528            [w0, w1, w2, w3, w4]
529        }
530
531        let r2: [u64; 5] = q3_times_n_keep_five(&q3);
532
533        #[inline]
534        #[allow(clippy::too_many_arguments)]
535        const fn sub_inner_five(l: [u64; 5], r: [u64; 5]) -> [u64; 5] {
536            let (w0, borrow) = sbb(l[0], r[0], 0);
537            let (w1, borrow) = sbb(l[1], r[1], borrow);
538            let (w2, borrow) = sbb(l[2], r[2], borrow);
539            let (w3, borrow) = sbb(l[3], r[3], borrow);
540            let (w4, _borrow) = sbb(l[4], r[4], borrow);
541
542            // If underflow occurred on the final limb - don't care (= add b^{k+1}).
543            [w0, w1, w2, w3, w4]
544        }
545
546        let r: [u64; 5] = sub_inner_five(r1, r2);
547
548        #[inline]
549        #[allow(clippy::too_many_arguments)]
550        const fn subtract_n_if_necessary(r0: u64, r1: u64, r2: u64, r3: u64, r4: u64) -> [u64; 5] {
551            let (w0, borrow) = sbb(r0, MODULUS[0], 0);
552            let (w1, borrow) = sbb(r1, MODULUS[1], borrow);
553            let (w2, borrow) = sbb(r2, MODULUS[2], borrow);
554            let (w3, borrow) = sbb(r3, MODULUS[3], borrow);
555            let (w4, borrow) = sbb(r4, 0, borrow);
556
557            // If underflow occurred on the final limb, borrow = 0xfff...fff, otherwise
558            // borrow = 0x000...000. Thus, we use it as a mask to conditionally add the
559            // modulus.
560            let (w0, carry) = adc(w0, MODULUS[0] & borrow, 0);
561            let (w1, carry) = adc(w1, MODULUS[1] & borrow, carry);
562            let (w2, carry) = adc(w2, MODULUS[2] & borrow, carry);
563            let (w3, carry) = adc(w3, MODULUS[3] & borrow, carry);
564            let (w4, _carry) = adc(w4, 0, carry);
565
566            [w0, w1, w2, w3, w4]
567        }
568
569        // Result is in range (0, 3*n - 1),
570        // and 90% of the time, no subtraction will be needed.
571        let r = subtract_n_if_necessary(r[0], r[1], r[2], r[3], r[4]);
572        let r = subtract_n_if_necessary(r[0], r[1], r[2], r[3], r[4]);
573        Scalar([r[0], r[1], r[2], r[3]])
574    }

576    /// Returns self * rhs mod n
577    pub const fn mul(&self, rhs: &Self) -> Self {
578        // Schoolbook multiplication.
579
580        let (w0, carry) = mac(0, self.0[0], rhs.0[0], 0);
581        let (w1, carry) = mac(0, self.0[0], rhs.0[1], carry);
582        let (w2, carry) = mac(0, self.0[0], rhs.0[2], carry);
583        let (w3, w4) = mac(0, self.0[0], rhs.0[3], carry);
584
585        let (w1, carry) = mac(w1, self.0[1], rhs.0[0], 0);
586        let (w2, carry) = mac(w2, self.0[1], rhs.0[1], carry);
587        let (w3, carry) = mac(w3, self.0[1], rhs.0[2], carry);
588        let (w4, w5) = mac(w4, self.0[1], rhs.0[3], carry);
589
590        let (w2, carry) = mac(w2, self.0[2], rhs.0[0], 0);
591        let (w3, carry) = mac(w3, self.0[2], rhs.0[1], carry);
592        let (w4, carry) = mac(w4, self.0[2], rhs.0[2], carry);
593        let (w5, w6) = mac(w5, self.0[2], rhs.0[3], carry);
594
595        let (w3, carry) = mac(w3, self.0[3], rhs.0[0], 0);
596        let (w4, carry) = mac(w4, self.0[3], rhs.0[1], carry);
597        let (w5, carry) = mac(w5, self.0[3], rhs.0[2], carry);
598        let (w6, w7) = mac(w6, self.0[3], rhs.0[3], carry);
599
600        Scalar::barrett_reduce(w0, w1, w2, w3, w4, w5, w6, w7)
601    }

604    pub const fn square(&self) -> Self {
605        // Schoolbook multiplication.
606        self.mul(self)
607    }

609    /// Returns `self^by`, where `by` is a little-endian integer exponent.
610    ///
611    /// **This operation is variable time with respect to the exponent.** If the exponent
612    /// is fixed, this operation is effectively constant time.
613    pub fn pow_vartime(&self, by: &[u64; 4]) -> Self {
614        let mut res = Self::one();
615        for e in by.iter().rev() {
616            for i in (0..64).rev() {
617                res = res.square();
618
619                if ((*e >> i) & 1) == 1 {
620                    res *= self;
621                }
622            }
623        }
624        res
625    }

627    /// Returns the multiplicative inverse of self, if self is non-zero
628    pub fn invert(&self) -> CtOption<Self> {
629        // We need to find b such that b * a ≡ 1 mod p. As we are in a prime
630        // field, we can apply Fermat's Little Theorem:
631        //
632        //    a^p         ≡ a mod p
633        //    a^(p-1)     ≡ 1 mod p
634        //    a^(p-2) * a ≡ 1 mod p
635        //
636        // Thus inversion can be implemented with a single exponentiation.
637        //
638        // This is `n - 2`, so the top right two digits are `4f` instead of `51`.
639        let inverse = self.pow_vartime(&[
640            0xf3b9_cac2_fc63_254f,
641            0xbce6_faad_a717_9e84,
642            0xffff_ffff_ffff_ffff,
643            0xffff_ffff_0000_0000,
644        ]);
645
646        CtOption::new(inverse, !self.is_zero())
647    }

663    /// Faster inversion using Stein's algorithm
664    pub fn invert_vartime(&self) -> CtOption<Self> {
665        // https://link.springer.com/article/10.1007/s13389-016-0135-4
666
667        let mut u = *self;
668        // currently an invalid scalar
669        let mut v = Scalar(MODULUS);
670
671        #[allow(non_snake_case)]
672        let mut A = Self::one();
673        #[allow(non_snake_case)]
674        let mut C = Self::zero();
675
676        while !bool::from(u.is_zero()) {
677            // u-loop
678            while bool::from(u.is_even()) {
679                u.shr1();
680                if bool::from(A.is_even()) {
681                    A.shr1();
682                } else {
683                    A.shr1();
684                    A += Scalar(MODULUS_SHR1);
685                    A += Self::one();
686                }
687            }
688
689            // v-loop
690            while bool::from(v.is_even()) {
691                v.shr1();
692                if bool::from(C.is_even()) {
693                    C.shr1();
694                } else {
695                    C.shr1();
696                    C += Scalar(MODULUS_SHR1);
697                    C += Self::one();
698                }
699            }
700
701            // sub-step
702            if u >= v {
703                u -= &v;
704                A -= &C;
705            } else {
706                v -= &u;
707                C -= &A;
708            }
709        }
710
711        CtOption::new(C, !self.is_zero())
712    }

862    fn from(scalar: &Scalar) -> ScalarBits {
863        let mut output = [0u32; 8];
864
865        for (input, output) in scalar.0.iter().zip(output.chunks_mut(2)) {
866            output[0] = (input & 0xFFFFFFFF) as u32;
867            output[1] = (input >> 32) as u32;
868        }
869
870        output.into()
871    }
