82bitflags::bitflags! {
83    #[derive(Debug, Eq, PartialEq, Clone, Copy)]
84    /// All non-used bits are RFU.
85    ///
86    /// In particular, top bit is intended to be used to accomodate breaking format changes,
87    /// i.e., if `flags >> 32 != 0`, then the format is different.
88    pub struct Flags: u16 {
89        /// Set if the key has been generated on this device.
90        const LOCAL = 1 << 0;
91        /// Set if the key is a secret key.
92        const SENSITIVE = 1 << 1;
93        // Reserved for future use
94        // const WRAPPABLE = 1 << 3;
95        /// This flag currently only applies to `kind::Shared`
96        const SERIALIZABLE = 1 << 4;
97    }
98}

110impl Key {
111    pub fn serialize(&self) -> SerializedKeyBytes {
112        let mut buffer = SerializedKeyBytes::new();
113        // big-endian here to ensure the first bit is enough to check compatibility
114        // on breaking format change
115        buffer
116            .extend_from_slice(&self.flags.bits().to_be_bytes())
117            .unwrap();
118        buffer
119            .extend_from_slice(&(self.kind.code()).to_be_bytes())
120            .unwrap();
121        // can't fail, since MAX_SERIALIZED_KEY_LENGTH is defined as MAX_KEY_MATERIAL_LENGTH + 4
122        buffer.extend_from_slice(&self.material).unwrap();
123        buffer
124    }
125
126    pub fn try_deserialize(bytes: &[u8]) -> Result<Self, Error> {
127        if bytes.len() < 4 {
128            return Err(Error::InvalidSerializedKey);
129        }
130        let (info, material) = bytes.split_at(4);
131        let flags_bits = u16::from_be_bytes([info[0], info[1]]);
132        let flags = Flags::from_bits(flags_bits).ok_or(Error::InvalidSerializedKey)?;
133
134        let kind_bits = u16::from_be_bytes([info[2], info[3]]);
135        let kind =
136            Kind::try_from(kind_bits, material.len()).map_err(|_| Error::InvalidSerializedKey)?;
137
138        Ok(Key {
139            flags,
140            kind,
141            material: Material::from_slice(material).map_err(|_| Error::InvalidSerializedKey)?,
142        })
143    }
144}

155impl Zeroize for Flags {
156    fn zeroize(&mut self) {
157        // Safety: We have a mutable reference, and Flags is `Copy` and therefore does not need to be dropped
158        unsafe {
159            write_volatile(self, Flags::empty());
160        }
161        atomic::compiler_fence(atomic::Ordering::SeqCst);
162    }

165/// Manual implementation to keep compatibility with version 0.1.0 that used `serde_indexed`
166/// serde_indexed cannot be used anymore for compatiblity with bitflags 2.0
167impl Serialize for Flags {
168    fn serialize<S>(&self, serializer: S) -> Result<S::Ok, S::Error>
169    where
170        S: serde::Serializer,
171    {
172        let mut map = serializer.serialize_map(Some(1))?;
173        map.serialize_key(&0usize)?;
174        map.serialize_value(&self.bits())?;
175        map.end()
176    }
177}

179/// Manual implementation to keep compatibility with version 0.1.0 that used `serde_indexed`
180impl<'de> Deserialize<'de> for Flags {
181    fn deserialize<D>(deserializer: D) -> Result<Self, D::Error>
182    where
183        D: serde::Deserializer<'de>,
184    {
185        struct FlagsVisitor;
186        impl<'vis_de> Visitor<'vis_de> for FlagsVisitor {
187            type Value = Flags;
188            fn expecting(&self, formatter: &mut core::fmt::Formatter) -> core::fmt::Result {
189                write!(formatter, "A flag structure")
190            }
191            fn visit_map<A>(self, mut map: A) -> Result<Self::Value, A::Error>
192            where
193                A: serde::de::MapAccess<'vis_de>,
194            {
195                if !matches!(map.next_key()?, Some(0usize)) {
196                    return Err(serde::de::Error::missing_field("bits"));
197                }
198                let bits = map.next_value()?;
199                let flags = Flags::from_bits(bits)
200                    .ok_or_else(|| serde::de::Error::custom("Wrong bit layout"))?;
201                Ok(flags)
202            }
203        }
204        deserializer.deserialize_map(FlagsVisitor)
205    }
206}

208impl Kind {
209    pub fn code(self) -> u16 {
210        match self {
211            Kind::Shared(_) => 1,
212            Kind::Symmetric(_) => 2,
213            Kind::Symmetric32Nonce(_) => 3,
214            Kind::Ed255 => 4,
215            Kind::P256 => 5,
216            Kind::X255 => 6,
217            Kind::Rsa2048 => 7,
218            Kind::Rsa3072 => 8,
219            Kind::Rsa4096 => 9,
220            Kind::P384 => 10,
221            Kind::P521 => 11,
222            Kind::BrainpoolP256R1 => 12,
223            Kind::BrainpoolP384R1 => 13,
224            Kind::BrainpoolP512R1 => 14,
225            Kind::Secp256k1 => 15,
226        }
227    }
228
229    pub fn try_from(code: u16, length: usize) -> Result<Self, Error> {
230        Ok(match code {
231            1 => Self::Shared(length),
232            2 => Self::Symmetric(length),
233            3 => Self::Symmetric32Nonce(length - 32),
234            4 => Self::Ed255,
235            5 => Self::P256,
236            6 => Self::X255,
237            7 => Kind::Rsa2048,
238            8 => Kind::Rsa3072,
239            9 => Kind::Rsa4096,
240            10 => Kind::P384,
241            11 => Kind::P521,
242            12 => Kind::BrainpoolP256R1,
243            13 => Kind::BrainpoolP384R1,
244            14 => Kind::BrainpoolP512R1,
245            15 => Kind::Secp256k1,
246            _ => return Err(Error::InvalidSerializedKey),
247        })
248    }
249}
