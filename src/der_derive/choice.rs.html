1//! Support for deriving the `Decodable` and `Encodable` traits on enums for
2//! the purposes of decoding/encoding ASN.1 `CHOICE` types as mapped to
3//! enum variants.
4
5use crate::{Asn1Attrs, Asn1Type};
6use proc_macro2::TokenStream;
7use quote::{quote, ToTokens};
8use syn::{DataEnum, Fields, FieldsUnnamed, Ident, Lifetime, Type, Variant};
9use synstructure::{Structure, VariantInfo};
10
11/// Registry of `CHOICE` alternatives for a given enum
12type Alternatives = std::collections::BTreeMap<Asn1Type, Alternative>;
13
14/// Derive the `Choice` trait for an enum.
15pub(crate) struct DeriveChoice {
16    /// `CHOICE` alternatives for this enum.
17    alternatives: Alternatives,
18
19    /// Tags included in the impl body for `der::Choice`.
20    choice_body: TokenStream,
21
22    /// Enum match arms for the impl body for `TryFrom<der::asn1::Any<'_>>`.
23    decode_body: TokenStream,
24
25    /// Enum match arms for the impl body for `der::Encodable::encode`.
26    encode_body: TokenStream,
27
28    /// Enum match arms for the impl body for `der::Encodable::encoded_len`.
29    encoded_len_body: TokenStream,
30}
31
32impl DeriveChoice {
33    /// Derive `Decodable` on an enum.
34    pub fn derive(s: Structure<'_>, data: &DataEnum, lifetime: Option<&Lifetime>) -> TokenStream {
35        assert_eq!(
36            s.variants().len(),
37            data.variants.len(),
38            "enum variant count mismatch"
39        );
40
41        let mut state = Self {
42            alternatives: Default::default(),
43            choice_body: TokenStream::new(),
44            decode_body: TokenStream::new(),
45            encode_body: TokenStream::new(),
46            encoded_len_body: TokenStream::new(),
47        };
48
49        for (variant_info, variant) in s.variants().iter().zip(&data.variants) {
50            let asn1_type = Asn1Attrs::new(&variant.attrs).asn1_type.unwrap_or_else(|| {
51                panic!(
52                    "no #[asn1(type=...)] specified for enum variant: {}",
53                    variant.ident
54                )
55            });
56
57            Alternative::register(&mut state.alternatives, asn1_type, variant);
58            state.derive_variant_choice(asn1_type);
59            state.derive_variant_decoder(asn1_type);
60
61            match variant_info.bindings().len() {
62                // TODO(tarcieri): handle 0 bindings for ASN.1 NULL
63                1 => {
64                    state.derive_variant_encoder(&variant_info, asn1_type);
65                    state.derive_variant_encoded_len(&variant_info);
66                }
67                other => panic!(
68                    "unsupported number of ASN.1 variant bindings for {}: {}",
69                    asn1_type, other
70                ),
71            }
72        }
73
74        state.finish(s, lifetime)
75    }
76
77    /// Derive the body of `Choice::can_decode
78    fn derive_variant_choice(&mut self, asn1_type: Asn1Type) {
79        let tag = asn1_type.tag();
80
81        if self.choice_body.is_empty() {
82            tag
83        } else {
84            quote!(| #tag)
85        }
86        .to_tokens(&mut self.choice_body);
87    }
88
89    /// Derive a match arm of the impl body for `TryFrom<der::asn1::Any<'_>>`.
90    fn derive_variant_decoder(&mut self, asn1_type: Asn1Type) {
91        let tag = asn1_type.tag();
92
93        let decoder = match asn1_type {
94            Asn1Type::BitString => quote!(any.bit_string()),
95            Asn1Type::GeneralizedTime => quote!(any.generalized_time()),
96            Asn1Type::OctetString => quote!(any.octet_string()),
97            Asn1Type::PrintableString => quote!(any.printable_string()),
98            Asn1Type::UtcTime => quote!(any.utc_time()),
99            Asn1Type::Utf8String => quote!(any.utf8_string()),
100        };
101
102        {
103            quote! {
104                #tag => #decoder
105                    .ok()
106                    .and_then(|val| val.try_into().ok())
107                    .ok_or_else(|| #tag.value_error()),
108            }
109        }
110        .to_tokens(&mut self.decode_body);
111    }
112
113    /// Derive a match arm for the impl body for `der::Encodable::encode`.
114    fn derive_variant_encoder(&mut self, variant: &VariantInfo<'_>, asn1_type: Asn1Type) {
115        assert_eq!(
116            variant.bindings().len(),
117            1,
118            "unexpected number of variant bindings"
119        );
120
121        variant
122            .each(|bi| {
123                let binding = &bi.binding;
124                let encoder_obj = asn1_type.encoder(quote!(#binding));
125                quote!(#encoder_obj?.encode(encoder))
126            })
127            .to_tokens(&mut self.encode_body);
128    }
129
130    /// Derive a match arm for the impl body for `der::Encodable::encode`.
131    fn derive_variant_encoded_len(&mut self, variant: &VariantInfo<'_>) {
132        assert_eq!(
133            variant.bindings().len(),
134            1,
135            "unexpected number of variant bindings"
136        );
137
138        variant
139            .each(|bi| {
140                let binding = &bi.binding;
141                quote!(#binding.encoded_len())
142            })
143            .to_tokens(&mut self.encoded_len_body);
144    }
145
146    /// Finish deriving an enum
147    fn finish(self, s: Structure<'_>, lifetime: Option<&Lifetime>) -> TokenStream {
148        let lifetime = match lifetime {
149            Some(lifetime) => quote!(#lifetime),
150            None => quote!('_),
151        };
152
153        let Self {
154            choice_body,
155            decode_body,
156            encode_body,
157            encoded_len_body,
158            ..
159        } = self;
160
161        let mut variant_conversions = TokenStream::new();
162
163        for variant in self.alternatives.values() {
164            let variant_ident = &variant.ident;
165            let variant_type = &variant.field_type;
166
167            variant_conversions.extend(s.gen_impl(quote! {
168                gen impl From<#variant_type> for @Self {
169                    fn from(field: #variant_type) -> Self {
170                        Self::#variant_ident(field)
171                    }
172                }
173            }));
174        }
175
176        s.gen_impl(quote! {
177            gen impl ::der::Choice<#lifetime> for @Self {
178                fn can_decode(tag: ::der::Tag) -> bool {
179                    matches!(tag, #choice_body)
180                }
181            }
182
183            gen impl core::convert::TryFrom<::der::asn1::Any<#lifetime>> for @Self {
184                type Error = der::Error;
185
186                fn try_from(any: ::der::asn1::Any<#lifetime>) -> der::Result<Self> {
187                    #[allow(unused_imports)]
188                    use core::convert::TryInto;
189
190                    match any.tag() {
191                        #decode_body
192                        actual => Err(der::ErrorKind::UnexpectedTag {
193                            expected: None,
194                            actual
195                        }
196                        .into()),
197                    }
198                }
199            }
200
201            gen impl ::der::Encodable for @Self {
202                fn encode(&self, encoder: &mut ::der::Encoder<'_>) -> ::der::Result<()> {
203                    #[allow(unused_imports)]
204                    use core::convert::TryFrom;
205
206                    match self {
207                        #encode_body
208                    }
209                }
210
211                fn encoded_len(&self) -> ::der::Result<::der::Length> {
212                    match self {
213                        #encoded_len_body
214                    }
215                }
216            }
217
218            #variant_conversions
219        })
220    }
221}

223/// ASN.1 `CHOICE` alternative: one of the ASN.1 types comprising the `CHOICE`
224/// which maps to an enum variant.
225struct Alternative {
226    /// [`Ident`] for the corresponding enum variant.
227    pub ident: Ident,
228
229    /// Type of the inner field (i.e. of the variant's 1-tuple)
230    pub field_type: Type,
231}
232
233impl Alternative {
234    /// Register a `CHOICE` alternative for a variant
235    pub fn register(alternatives: &mut Alternatives, asn1_type: Asn1Type, variant: &Variant) {
236        let field_type = match &variant.fields {
237            Fields::Unnamed(FieldsUnnamed { unnamed, .. }) if unnamed.len() == 1 => {
238                let field = unnamed.first().unwrap();
239                field.ty.clone()
240            }
241            _ => panic!("can only derive `Choice` for enums with 1-tuple variants"),
242        };
243
244        let alternative = Self {
245            ident: variant.ident.clone(),
246            field_type,
247        };
248
249        if let Some(duplicate) = alternatives.insert(asn1_type, alternative) {
250            panic!(
251                "duplicate ASN.1 type `{}` for enum variants `{}` and `{}`",
252                asn1_type, duplicate.ident, variant.ident
253            );
254        }
255    }
256}
